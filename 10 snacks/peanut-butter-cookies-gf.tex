\section*{Directions}

\noindent
Preheat oven to \textit{350°F} ---
Line a baking sheet with parchment ---
Have 1~Tbsp. \textbf{sugar} for sprinkling in \textit{Small Bowl~\#1}

\begin{enumerate}
    \item In \textit{Medium Bowl~\#1}, combine 1~cup \textbf{peanut butter}, 1~large \textbf{egg}, 1~cup \textbf{sugar}, 1~tsp. \textbf{baking soda}, 1~tsp. \textbf{Mexican vanilla}, and \nicefrac{1}{4}~tsp. \textbf{table salt}. Mix with an electric paddle or wooden spoon until evenly combined and no streaks remain.

    \item Portion dough with a 1\nicefrac{3}{4}-inch cookie scoop into 12 balls and place about 2 inches apart on the prepared sheet.

    \item Sprinkle the tops with \textbf{sugar} (\textit{Small Bowl~\#1}), then press each ball with a fork to make a crosshatch pattern.

    \item Bake at \textit{350°F} for \textit{11--12~minutes} until cookies are set and lightly golden at the edges; centers may still look soft but will firm as they cool. Continue baking in \textit{1~minute} increments if needed.

    \item Cool on the sheet for \textit{2--3~minutes}, then transfer to a wire rack to cool completely.
\end{enumerate}

\subsection*{Yield}
\begin{itemize}
    \item Makes 12 cookies
\end{itemize}

\subsection*{Equipment Required}
\begin{itemize}
    \item Rimmed baking sheet
    \item Parchment paper
    \item Medium mixing bowl
    \item Small bowl (for sugar for sprinkling)
    \item 1\nicefrac{3}{4}-inch cookie scoop
    \item Fork (for hatch marks)
    \item Electric mixer with paddle attachment or wooden spoon
    \item Measuring cups and spoons
    \item Wire rack (optional, for cooling)
\end{itemize}

\subsection*{Mise en Place}
\begin{itemize}
    \item \textit{Medium Bowl~\#1} --- dough: 1~cup \textbf{peanut butter}, 1~large \textbf{egg}, 1~cup \textbf{sugar}, 1~tsp. \textbf{baking soda}, 1~tsp. \textbf{Mexican vanilla}, \nicefrac{1}{4}~tsp. \textbf{table salt}
    \item Preheat oven and line sheet with parchment before mixing
\end{itemize}

\subsection*{Ingredient Tips}
\begin{itemize}
    \item Use creamy \textbf{peanut butter} (standard jar, not natural/oil-separating) for consistent texture
    \item \textbf{Mexican vanilla} adds warm, rounded flavor; pure vanilla extract may be substituted
    \item \textbf{Table salt} dissolves evenly in the dough; avoid coarse salt
\end{itemize}

\subsection*{Preparation Tips}
\begin{itemize}
    \item Dough is soft and sticky; the scoop keeps portions even and avoids overhandling
    \item Sprinkle \textbf{sugar} before pressing with the fork so it adheres to the tops
    \item Cookies may look underdone at \textit{11--12~minutes}; edges should be lightly golden and centers will set as they cool
    \item Don't overbake---cookies will become dry and crumbly
\end{itemize}

\subsection*{Make Ahead \& Storage}
\begin{itemize}
    \item Store in an airtight container at room temperature for up to \textit{3--5~days}
    \item Dough can be portioned and refrigerated up to \textit{2~days}; bake from cold and add 1--2 minutes if needed
    \item Freeze baked cookies in an airtight container for up to \textit{2~months}; thaw at room temperature
\end{itemize}

\subsection*{Serving Suggestions}
\begin{itemize}
    \item Serve with cold milk or coffee
    \item Pair with vanilla ice cream for a simple dessert
    \item Pack for lunchboxes or picnics
\end{itemize}
