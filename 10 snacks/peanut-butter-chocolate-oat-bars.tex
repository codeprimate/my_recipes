\section*{Oat Crust}

\noindent
Preheat oven to \textit{325°F} ---
Line 8-inch×8-inch pan with parchment paper, leaving overhang on two sides ---
Bring 6~Tbsp. \textbf{butter} to room temperature for filling

\begin{enumerate}
    \item In \textit{Medium Bowl~\#1}, whisk together 1\nicefrac{1}{2}~cups \textbf{rolled oats}, \nicefrac{1}{4}~cup \textbf{GF flour}, \nicefrac{1}{4}~cup \textbf{brown sugar}, \nicefrac{1}{4}~tsp. \textbf{salt}, and \nicefrac{1}{4}~tsp. \textbf{cinnamon}. Add 5~Tbsp. melted \textbf{butter}, 1~Tbsp. \textbf{honey}, and \nicefrac{1}{4}~tsp. \textbf{Mexican vanilla}. Stir until evenly combined and mixture holds together when pressed.
    
    \item Press mixture firmly and evenly into the prepared pan, using the bottom of a measuring cup or glass to compact into a uniform layer.
    
    \item Bake at \textit{325°F} for \textit{12--15~minutes} until edges are lightly golden and crust is fragrant. Remove from oven and let cool completely in pan, about \textit{30~minutes}.
\end{enumerate}

\section*{Peanut Butter Filling}

\begin{enumerate}
    \item In \textit{Medium Bowl~\#2}, combine 1~cup \textbf{peanut butter} and 6~Tbsp. softened \textbf{butter}. Beat with a spatula or hand mixer until smooth and creamy.
    
    \item Add 1\nicefrac{3}{4}~cups \textbf{powdered sugar} and \nicefrac{1}{2}~tsp. \textbf{Mexican vanilla}. Mix until fully incorporated and smooth, scraping down sides as needed.
    
    \item Spread filling evenly over the cooled \textbf{oat crust}, using an offset spatula or the back of a spoon to create a smooth, level layer.
    
    \item Refrigerate for \textit{30~minutes} until filling is firm to the touch.
\end{enumerate}

\section*{Chocolate Topping}

\begin{enumerate}
    \item In \textit{Medium Bowl~\#3}, combine \nicefrac{3}{4}~cup \textbf{semi-sweet chocolate chips}, \nicefrac{1}{4}~cup \textbf{dark chocolate chips}, and 1~Tbsp. \textbf{peanut butter}.
    
    \item Microwave \textit{Medium Bowl~\#3} in \textit{30-second} intervals, stirring after each, until chocolate is melted and smooth. Alternatively, melt in a double boiler over barely simmering water.
    
    \item Pour melted \textbf{chocolate} over the chilled \textbf{peanut butter} layer, spreading evenly to the edges with an offset spatula.
    
    \item Immediately sprinkle \textbf{honey roasted peanuts} (\textit{Small Bowl~\#2}) evenly over the \textbf{chocolate}, pressing gently so they adhere.
    
    \item Refrigerate for at least \textit{1~hour} until chocolate is fully set.
\end{enumerate}

\section*{Cutting and Serving}

\begin{enumerate}
    \item Use parchment overhang to lift bars from pan onto a cutting board.
    
    \item Let stand at room temperature for \textit{5~minutes} before cutting---this prevents chocolate from cracking.
    
    \item Using a sharp knife, cut into 16 squares (4×4 grid), wiping blade clean between cuts for neat edges.
\end{enumerate}

\subsection*{Equipment Required}
\begin{itemize}
    \item 8×8-inch baking pan
    \item Parchment paper
    \item Medium mixing bowls (3)
    \item Measuring cups and spoons
    \item Flat-bottomed measuring cup or glass (for pressing crust)

\end{itemize}

\subsection*{Mise en Place}
\begin{itemize}
    \item \textit{Medium Bowl~\#1} --- \textbf{oat} crust mixture: 1\nicefrac{1}{2}~cups \textbf{oats}, \nicefrac{1}{4}~cup \textbf{GF flour}, 5~Tbsp. melted \textbf{butter}, \nicefrac{1}{4}~cup \textbf{brown sugar}, 1~Tbsp. \textbf{honey}, \nicefrac{1}{4}~tsp. \textbf{salt}, \nicefrac{1}{4}~tsp. \textbf{cinnamon}, \nicefrac{1}{4}~tsp. \textbf{Mexican vanilla}
    \item \textit{Medium Bowl~\#2} --- \textbf{peanut butter} filling: 1~cup \textbf{peanut butter}, 6~Tbsp. softened \textbf{butter}, 1\nicefrac{3}{4}~cups \textbf{powdered sugar}, \nicefrac{1}{2}~tsp. \textbf{Mexican vanilla}
    \item \textit{Medium Bowl~\#3} --- \textbf{chocolate} topping: \nicefrac{3}{4}~cup \textbf{semi-sweet chips}, \nicefrac{1}{4}~cup \textbf{dark chips}, 1~Tbsp. \textbf{peanut butter}
    \item \textit{Small Bowl~\#2} --- \nicefrac{1}{3}~cup chopped \textbf{honey roasted peanuts}
    \item Bring 6~Tbsp. \textbf{butter} to room temperature for filling while crust bakes and cools
    \item Line pan with parchment before starting
\end{itemize}

\subsection*{Ingredient Tips}
\begin{itemize}
    \item Use \textbf{old fashioned rolled oats}---quick oats become pasty; steel-cut won't bind
    \item \textbf{1:1 GF flour} (Bob's Red Mill or King Arthur Measure for Measure) provides structure; regular all-purpose flour works equally well if gluten is not a concern
    \item Standard creamy \textbf{peanut butter} (Jif, Skippy) works best; natural peanut butter with oil separation may not set properly
    \item Salted \textbf{butter} throughout adds flavor depth; no additional finishing salt needed
    \item \textbf{Semi-sweet} and \textbf{dark chocolate} blend provides richness without excessive sweetness
    \item \textbf{Honey roasted peanuts} add sweetness, salt, and textural contrast---chop coarsely for visible pieces
    \item Sift \textbf{powdered sugar} if lumpy for smoothest filling
\end{itemize}

\subsection*{Preparation Tips}
\begin{itemize}
    \item Press \textbf{oat} crust firmly and evenly---the \textbf{honey} and \textbf{flour} help it hold together, but loose spots will still crumble
    \item Cool crust completely before adding filling; warm crust will melt the filling unevenly
    \item \textbf{Peanut butter} filling should be thick but spreadable---if too stiff, let butter soften more
    \item Chill filling until firm before adding \textbf{chocolate}; soft filling causes chocolate to sink
    \item Don't overheat \textbf{chocolate}---stir frequently and remove as soon as smooth
    \item Add 1~Tbsp. \textbf{peanut butter} to chocolate for authentic Reese's flavor and easier cutting
    \item Press \textbf{peanuts} gently into chocolate before it sets
    \item Let bars temper at room temperature before cutting to prevent chocolate cracking
    \item Wipe knife between cuts for clean edges
\end{itemize}

\subsection*{Make Ahead \& Storage}
\begin{itemize}
    \item Store covered in refrigerator for up to \textit{1~week}
    \item Bars are best served slightly chilled or at cool room temperature
    \item For room temperature serving, let stand \textit{10--15~minutes} after removing from refrigerator
    \item Can be frozen in airtight container for up to \textit{2~months}
    \item Thaw frozen bars in refrigerator overnight
    \item Layer bars between parchment when stacking for storage
\end{itemize}

\subsection*{Serving Suggestions}
\begin{itemize}
    \item Serve as a rich dessert or sweet snack
    \item Pairs well with cold milk, coffee, or vanilla ice cream
    \item Cut into smaller pieces (5×5 grid = 25 pieces) for party bites
    \item Excellent for potlucks, bake sales, and gift giving
    \item Store at room temperature for softer texture, refrigerate for firmer, snappier chocolate
\end{itemize}
