\section*{Ingredients}
\setlength{\columnsep}{20pt}
\begin{multicols}{2}
\noindent
    Popcorn \dotfill \nicefrac{1}{3} cup \\
	Coconut oil \dotfill 2 Tbsp. \\
	Turbinado sugar \dotfill 3 Tbsp. \\
	Mexican vanilla \dotfill \nicefrac{1}{4} tsp. \\
	Butter \dotfill \nicefrac{1}{2} Tbsp. \\
	Salt \dotfill \nicefrac{1}{4}-\nicefrac{1}{2} tsp.
\end{multicols}

\section*{Directions}

\begin{enumerate}
	\item Add \textbf{coconut oil} and \nicefrac{1}{4}~tsp. \textbf{vanilla} to popcorn maker plate.
	\item Sprinkle in \nicefrac{1}{3}~cup \textbf{popcorn} and 3~Tbsp. \textbf{turbinado sugar}.
	\item Thinly slice \nicefrac{1}{2}~Tbsp. \textbf{butter} and place on dripping shelf.
	\item Turn on popcorn maker.
	\item Two or three times during popping, gently shake the popcorn maker side to side to dislodge stuck kernels. \textbf{TAKE CARE!}
	\item When the pops slow to every \textit{3~seconds}, quickly turn off power and transfer popcorn to a large bowl.
	\item Salt to taste with \nicefrac{1}{4}-\nicefrac{1}{2}~tsp. \textbf{salt}.
\end{enumerate}
