\section*{Roasted Hatch Green Chiles}
Preheat air fryer to \textit{400°F}

\begin{enumerate}
    \item Wash and dry the \textbf{Hatch green chiles}.
    \item Place \textbf{chiles} in the air fryer basket in a single layer, leaving space between each chile.
    \item Air fry at \textit{400°F} for \textit{5-7 minutes}, then flip and cook for another \textit{5-7 minutes} until the skin is blistered and charred.
    \item Remove from the air fryer and place in a bowl covered with plastic wrap to steam for \textit{10-15 minutes}.
    \item Once cooled, peel off the skin, remove seeds, and chop the \textbf{chiles}.
\end{enumerate}

\section*{Hatch Green Chile Chicken Enchiladas}

\noindent
Preheat oven to \textit{350°F} ---
Grease a 9~inch×13~inch baking dish ---
Shred \textbf{Monterey Jack cheese}; set aside in \textit{Large Bowl~\#1} ---
Dice \textbf{onion}; set aside in \textit{Small Bowl~\#1} ---
Mince \textbf{garlic}; set aside in \textit{Small Bowl~\#2} ---
Chop \textbf{cilantro}; set aside in \textit{Small Bowl~\#3} ---
Juice \textbf{lime}; set aside in \textit{Small Bowl~\#4}

\begin{enumerate}
    \item In a large skillet, heat 2~Tbsp. \textbf{vegetable oil} over medium-high heat. Season \textbf{chicken breasts} with \textbf{salt} and \textbf{pepper}, then cook until golden brown and cooked through, about \textit{6-8~minutes} per side. Remove from heat, let cool, then shred the \textbf{chicken}; set aside in \textit{Large Bowl~\#2}.
    
    \item In the same skillet, sauté \textbf{onion} (\textit{Small Bowl~\#1}) until translucent, about \textit{5~minutes}. Add \textbf{garlic} (\textit{Small Bowl~\#2}) and cook for another \textit{1~minute}. Add 1~cup of chopped \textbf{roasted Hatch green chiles}, 2~tsp. \textbf{cumin}, and 1~tsp. \textbf{oregano}. Cook for \textit{2-3~minutes} until fragrant.
    
    \item Add shredded \textbf{chicken} (\textit{Large Bowl~\#2}) to the skillet and mix well. Season with \textbf{salt} and \textbf{pepper} to taste. Transfer to \textit{Large Bowl~\#2}; set aside.
    
    \item In a saucepan, melt \nicefrac{1}{4}~cup \textbf{butter} over medium heat. Whisk in \nicefrac{1}{4}~cup \textbf{flour} and cook for \textit{1-2~minutes}. Gradually whisk in 1~cup \textbf{chicken broth}, 1~cup \textbf{heavy cream}, and 1~cup \textbf{sour cream}. Bring to a simmer and cook until thickened, about \textit{5~minutes}. Stir in remaining 1~cup of \textbf{roasted Hatch green chiles}, \textbf{lime juice} (\textit{Small Bowl~\#4}), and \textbf{cilantro} (\textit{Small Bowl~\#3}). Season with \textbf{salt} and \textbf{pepper} to taste. Transfer to \textit{Medium Bowl~\#1}; set aside.
    
    \item Warm \textbf{corn tortillas} in the microwave or on a griddle until pliable. Dip each \textbf{tortilla} in the sauce (\textit{Medium Bowl~\#1}), then fill with the \textbf{chicken} mixture (\textit{Large Bowl~\#2}) and some \textbf{Monterey Jack cheese} (\textit{Large Bowl~\#1}). Roll up and place seam-side down in the prepared baking dish.
    
    \item Pour the remaining sauce (\textit{Medium Bowl~\#1}) over the \textbf{enchiladas} and sprinkle with remaining \textbf{Monterey Jack cheese} (\textit{Large Bowl~\#1}).
    
    \item Bake for \textit{25-30 minutes}, until the \textbf{cheese} is melted and bubbly.
    
    \item Garnish with additional \textbf{cilantro}, if desired. Serve hot with your favorite sides like \textbf{Mexican rice}, \textbf{refried beans}, or a fresh salad.
\end{enumerate}
