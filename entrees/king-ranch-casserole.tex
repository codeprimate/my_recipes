\section*{Ingredients}

\noindent
    Chicken thighs, bone-in skin-on \dotfill 3 lbs \\
    Water \dotfill 2 cups \\
    Mexican spice mix \dotfill 3 Tbsp. \\
    Caldo de tomate bouillon \dotfill 1 Tbsp. \\
    Better Than Bouillon - Chicken \dotfill 1 Tbsp. \\
    Butter \dotfill 4 Tbsp. \\
    Flour \dotfill \nicefrac{1}{4} cup \\
    Heavy cream \dotfill 1\nicefrac{1}{2} cups \\
    Mushrooms, canned \dotfill 8 oz. can \\
    Garlic cloves \dotfill 4-6 \\
    \columnbreak

    White pepper \dotfill \nicefrac{1}{4} tsp. \\
    Ground nutmeg \dotfill \nicefrac{1}{8} tsp. \\
    MSG \dotfill \nicefrac{1}{2} tsp. \\
    Lard \dotfill 2 Tbsp. + 1 tsp. \\
    Onion \dotfill 1 large \\
    Bell pepper \dotfill 1 \\
    Ro-Tel \dotfill 10-14 oz. can \\
    Corn tortillas, medium \dotfill 14 \\
    Sharp cheddar cheese \dotfill 16 oz. \\
\end{multicols}

\section*{Directions}

\noindent
Preheat oven to \textit{375°F} ---
Bring \textbf{chicken thighs} to room temperature ---
Quarter \textbf{corn tortillas}; set aside in \textit{Medium Bowl~\#1} ---
Grease a 9~inch×13~inch baking dish with 1~tsp. \textbf{lard} ---
Dice \textbf{onion} and \textbf{bell pepper}; combine in \textit{Large Bowl~\#1} ---
Drain \textbf{mushrooms} and pat dry with paper towels; set aside in \textit{Medium Bowl~\#2} ---
Mince \textbf{garlic}; set aside in \textit{Small Bowl~\#1} ---
Combine \textbf{Mexican spice mix}, \textbf{caldo de tomate}, and \textbf{bouillon} in \textit{Small Bowl~\#2} (spice blend) ---
Combine \textbf{onion powder}, \textbf{white pepper}, \textbf{nutmeg}, and \textbf{MSG} in \textit{Small Bowl~\#3} (sauce spices) ---
Shred \textbf{cheese}; set aside in \textit{Large Bowl~\#2}

\begin{enumerate}
    \item Add 2~cups \textbf{water} to the pressure cooker and heat on \textit{simmer} mode. Stir in \textbf{spice blend} (\textit{Small Bowl~\#2}) until dissolved.
    
    \item Add the \textbf{chicken thighs} to the \textbf{seasoned broth}.
    
    \item Cook on \textit{high} pressure for \textit{14~minutes}, then allow for natural release for \textit{5~minutes}.
    
    \item Remove the \textbf{chicken thighs}, discard skin and bones, and shred the meat. Strain the \textbf{cooking~broth} through a fine-mesh sieve and reserve 1~cup. Set \textbf{shredded chicken} aside.
    
    \item In a large saucepan, melt 4~Tbsp. \textbf{butter} over medium heat. Add \textbf{mushrooms} (\textit{Medium Bowl~\#2}) and sauté until lightly browned, about \textit{3-4~minutes}. Add \textbf{garlic} (\textit{Small Bowl~\#1}) and cook for \textit{1~minute}.
    
    \item Sprinkle \nicefrac{1}{4}~cup \textbf{flour} over the mushroom mixture and cook, stirring constantly, for \textit{2~minutes}.
    
    \item Gradually whisk in 1\nicefrac{1}{2}~cups \textbf{heavy cream} and reserved \textbf{cooking broth}. Add \textbf{sauce spices} (\textit{Small Bowl~\#3}). Simmer, stirring frequently, until thickened, about \textit{5~minutes}. Keep warm.
    
    \item In a large skillet, melt 2~Tbsp. \textbf{lard} over medium heat. Sauté \textbf{onion} and \textbf{bell pepper} (\textit{Large Bowl~\#1}) for \textit{5~minutes}. Raise heat to \textit{high} and lightly scorch for \textit{1-2~minutes}.
    
    \item Reduce heat to \textit{medium}. Add \textbf{cream sauce} from the saucepan and \textbf{Ro-Tel}. Stir to combine.
    
    \item Bring to a simmer, reduce heat to \textit{medium-low}, and cook, stirring occasionally, for \textit{5~minutes}.
    
    \item Add the \textbf{shredded chicken} and stir to coat evenly. Remove from heat.
    
    \item Layer ingredients in the prepared baking dish:
    \begin{itemize}
        \item \nicefrac{1}{3}~of the \textbf{tortilla quarters} (\textit{Medium Bowl~\#1})
        \item \nicefrac{1}{2}~of the \textbf{chicken mixture}
        \item \nicefrac{1}{3}~of the \textbf{shredded cheese} (\textit{Large Bowl~\#2})
    \end{itemize}
    Repeat layers, then top with remaining \textbf{tortillas} and \textbf{cheese}.
    
    \item Bake uncovered for \textit{30-35~minutes} until bubbly and cheese is melted.
    
    \item Let stand for \textit{10~minutes} before serving.
\end{enumerate}

\subsection*{Equipment Required}
\begin{itemize}
    \item Pressure cooker or Instant Pot
    \item 9x13 inch baking dish
    \item Large skillet (12-inch preferred)
    \item Large saucepan for cream sauce
    \item 3 small prep bowls
    \item 2 medium prep bowls
    \item 2 large prep bowls
    \item Measuring cups and spoons
    \item Whisk
    \item Wooden spoon or spatula
    \item Sharp knife and cutting board
    \item Box grater for cheese
    \item Fine-mesh strainer or sieve
    \item Can opener
    \item Paper towels (for drying mushrooms)
    \item Colander (optional, for draining mushrooms and Ro-Tel)
\end{itemize}

\subsection*{Mise en Place}
\begin{itemize}
    \item Bring \textbf{chicken thighs} and \textbf{cream} to room temperature (\textit{30~minutes})
    \item \textit{Small Bowl~\#1} --- minced \textbf{garlic}
    \item \textit{Small Bowl~\#2} --- \textbf{spice blend}: 3~Tbsp. \textbf{Mexican~spice~mix}, 1~Tbsp. \textbf{caldo~de~tomate}, 1~Tbsp. \textbf{bouillon}
    \item \textit{Small Bowl~\#3} --- \textbf{sauce spices}: 1~tsp. \textbf{onion~powder}, \nicefrac{1}{4}~tsp. \textbf{white~pepper}, \nicefrac{1}{8}~tsp. \textbf{nutmeg}, \nicefrac{1}{2}~tsp. \textbf{MSG}
    \item \textit{Medium Bowl~\#1} --- quartered \textbf{corn tortillas} (14~tortillas)
    \item \textit{Medium Bowl~\#2} --- drained and patted dry \textbf{canned mushrooms} (8~oz. can)
    \item \textit{Large Bowl~\#1} --- diced \textbf{onion} and \textbf{bell pepper}
    \item \textit{Large Bowl~\#2} --- shredded \textbf{cheddar cheese} (16~oz.)
    \item Cut and prep all vegetables before starting
\end{itemize}

\subsection*{Ingredient Tips}
\begin{itemize}
    \item Use cup4cup or King Arthur Measure for Measure gluten-free flour for best roux results
    \item Drain \textbf{canned mushrooms} thoroughly and pat dry to remove excess moisture before sautéing
    \item Extra-sharp cheddar adds more depth of flavor
    \item If \textbf{Mexican~spice~mix} unavailable, combine equal parts ground cumin, chili powder, and paprika
    \item Authentic \textbf{lard} makes a difference - avoid hydrogenated versions
\end{itemize}

\subsection*{Preparation Tips}
\begin{itemize}
    \item Cook roux until it smells nutty but watch carefully as gluten-free flour can burn quickly
    \item Don't skip scorching the vegetables - it adds essential flavor
    \item Pat \textbf{canned mushrooms} very dry before sautéing to help them brown
    \item Keep sauce warm until assembly to maintain proper consistency
    \item Warm tortillas slightly before cutting to prevent cracking
\end{itemize}

\subsection*{Make Ahead \& Storage}
\begin{itemize}
    \item Chicken can be cooked and shredded up to \textit{2~days} ahead
    \item Complete sauce up to \textit{24~hours} in advance
    \item Assembled casserole can be refrigerated up to \textit{24~hours} before baking
    \item If made ahead, add \textit{10~minutes} to baking time
    \item Leftovers keep for \textit{3~days}; reheat covered at \textit{350°F}
\end{itemize}

\subsection*{Serving Suggestions}
\begin{itemize}
    \item Serve with Mexican rice and refried beans
    \item Garnish with diced avocado and fresh cilantro
    \item Offer sour cream and hot sauce on the side
    \item Can be frozen in individual portions for \textit{2~months}
\end{itemize}
