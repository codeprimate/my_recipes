\section*{Directions}

\noindent
Preheat oven to \textit{375°F} ---
Bring \textbf{chicken thighs} and \textbf{heavy cream} to room temperature (\textit{30~minutes}) ---
Combine \textbf{Mexican spice mix}, \textbf{caldo de tomate}, and \textbf{bouillon} in \textit{Small Bowl~\#1} (spice blend) ---
Combine \textbf{onion powder}, \textbf{white pepper}, \textbf{nutmeg}, and \textbf{MSG} in \textit{Small Bowl~\#2} (sauce spices) ---
Quarter 16 \textbf{corn tortillas}; set aside on a plate ---
Drain \textbf{mushrooms} and \textbf{sweet corn} and pat dry with paper towels; combine in \textit{Medium Bowl~\#1} ---
Dice \textbf{onion} and \textbf{bell pepper}; mince \textbf{garlic}; combine in \textit{Large Bowl~\#1} ---
Shred \textbf{cheese}; set aside in \textit{Large Bowl~\#2} ---
Grease a 3 quart baking dish with 1~tsp. \textbf{lard}

\begin{enumerate}
    \item Add \textbf{spice blend} (\textit{Small Bowl~\#1}) to the Instant Pot and heat on \textit{sauté} mode. Toast spices, stirring constantly, for \textit{1~minute}. Add 2~cups \textbf{water} and stir until dissolved.
    
    \item Add the \textbf{chicken thighs} to the \textbf{seasoned broth}.
    
    \item Cook on \textit{high} pressure for \textit{14~minutes}, then allow for natural release for \textit{5~minutes}.
    
    \item Transfer \textbf{chicken thighs} to a plate and let rest for \textit{5-10~minutes} until cool enough to handle.
    Meanwhile, skim the fat from the \textbf{stock} and strain the remaining stock through a fine-mesh sieve and reserve 1\nicefrac{1}{2} cups in \textit{Small Bowl~\#3} for the cream sauce.

    \item When chicken is cool, discard skin and bones, shred the meat into \textit{Large Bowl~\#3}. Skim any additional fat from the stock and add the remaining stock to the bowl and stir to combine.
    
    \item In a large skillet, melt 2~Tbsp. \textbf{lard} over medium heat. Add \textbf{mushrooms} and \textbf{sweet corn} (\textit{Medium Bowl~\#1}) and sauté until well browned, about \textit{5-6~minutes}.
    
    \item Add \textbf{onion}, \textbf{bell pepper}, and \textbf{garlic} (\textit{Large Bowl~\#1}). Sauté for \textit{8~minutes}. Raise heat to \textit{high} and cook undisturbed for \textit{1~minute}, then stir and cook \textit{1~minute} more until vegetables have dark caramelized spots on edges.
    
    \item Meanwhile, in a small saucepan, melt 4~Tbsp. \textbf{butter} over medium heat. Sprinkle in 2~Tbsp. \textbf{flour} and cook, stirring constantly, until the mixture is smooth and smells nutty, about \textit{2~minutes}. The roux should be pale golden.
    
    \item Gradually whisk in 1\nicefrac{1}{2}~cups \textbf{heavy cream} and 1\nicefrac{1}{2} cups reserved \textbf{cooking broth} (\textit{Small Bowl~\#3}). Add \textbf{sauce spices} (\textit{Small Bowl~\#2}). Simmer, stirring frequently, until thickened to coat the back of a spoon, about \textit{5~minutes}.
    
    \item Reduce skillet heat to \textit{medium}. Add \textbf{cream sauce} from the saucepan and undrained \textbf{Ro-Tel}. Stir to combine. 
    
    \item Bring to a simmer, reduce heat to \textit{medium-low}, and cook, stirring occasionally, for \textit{5~minutes}.
    
    \item Pour \textbf{vegetable and sauce mixture} from the skillet over the \textbf{shredded chicken} (\textit{Large Bowl~\#3}) and stir to coat evenly.
    
    \item Layer ingredients in the prepared baking dish:
    \begin{itemize}
        \item First layer: \nicefrac{1}{3}~of the \textbf{tortilla quarters}, \nicefrac{1}{2}~of the \textbf{chicken mixture}, \nicefrac{1}{3}~of the \textbf{shredded cheese} (\textit{Large Bowl~\#2})
        \item Second layer: \nicefrac{1}{3}~of the \textbf{tortilla quarters}, remaining \textbf{chicken mixture}, \nicefrac{1}{3}~of the \textbf{shredded cheese} (\textit{Large Bowl~\#2})
        \item Final layer: remaining \textbf{tortilla quarters}, remaining \textbf{shredded cheese} (\textit{Large Bowl~\#2})
    \end{itemize}
    
    \item Bake uncovered for \textit{30-35~minutes} until bubbly and cheese is melted.
    
    \item Let stand for \textit{10~minutes} before serving.
\end{enumerate}

\subsection*{Equipment Required}
\begin{itemize}
    \item Instant Pot or pressure cooker
    \item 3 Quart (9x13) inch baking dish
    \item Large skillet (deep 12-inch preferred)
    \item Small saucepan for cream sauce
    \item 3 small prep bowls
    \item 1 medium prep bowl
    \item 3 large prep bowls
    \item 2 Large plates (for tortillas and resting chicken)
    \item Measuring cups and spoons
    \item Whisk
    \item Wooden spoon or spatula
    \item Sharp knife and cutting board
    \item Box grater for cheese
    \item Fine-mesh strainer or sieve
    \item Can opener
    \item Paper towels (for drying mushrooms)
\end{itemize}

\subsection*{Mise en Place}
\begin{itemize}
    \item Bring \textbf{chicken thighs} and \textbf{cream} to room temperature (\textit{30~minutes})
    \item \textit{Small Bowl~\#1} --- \textbf{spice blend}: 4~Tbsp. \textbf{Mexican~spice~mix}, 1~Tbsp. \textbf{caldo~de~tomate}, 1~Tbsp. \textbf{bouillon}
    \item \textit{Small Bowl~\#2} --- \textbf{sauce spices}: 1~tsp. \textbf{onion~powder}, \nicefrac{1}{4}~tsp. \textbf{white~pepper}, \nicefrac{1}{8}~tsp. \textbf{nutmeg}, \nicefrac{1}{2}~tsp. \textbf{MSG}
    \item \textit{Small Bowl~\#3} --- reserved for strained \textbf{cooking broth} (1\nicefrac{1}{2}~cups for cream sauce)
    \item \textit{Large Plate} --- quartered \textbf{corn tortillas} (16~tortillas)
    \item \textit{Medium Bowl~\#1} --- drained and patted dry \textbf{canned mushrooms} (8~oz. can) and \textbf{sweet corn} (10~oz. can)
    \item \textit{Large Bowl~\#1} --- diced \textbf{onion}, diced \textbf{bell pepper}, and minced \textbf{garlic}
    \item \textit{Large Bowl~\#2} --- shredded \textbf{cheddar cheese} (16~oz.)
    \item \textit{Large Bowl~\#3} --- reserved for \textbf{shredded chicken} (after cooking), will become final mixing bowl
    \item Keep \textbf{Ro-Tel} undrained
    \item Cut and prep all vegetables before starting
\end{itemize}

\subsection*{Ingredient Tips}
\begin{itemize}
    \item Drain \textbf{canned mushrooms} thoroughly and pat dry to remove excess moisture before sautéing
    \item Extra-sharp cheddar adds more depth of flavor
    \item If \textbf{Mexican~spice~mix} unavailable, combine equal parts ground cumin, chili powder, and paprika
    \item Authentic \textbf{lard} makes a difference - avoid hydrogenated versions
\end{itemize}

\subsection*{Preparation Tips}
\begin{itemize}
    \item Pat \textbf{canned mushrooms} very dry before sautéing to help them brown deeply
    \item Cook \textbf{mushrooms} first in the skillet to develop deep browning without crowding
    \item Make the cream sauce in a separate small saucepan while vegetables cook in the skillet
    \item Cook roux until it smells nutty and is pale golden, but watch carefully to avoid burning
    \item Don't skip charring the vegetables - look for dark caramelized spots on edges for essential depth of flavor
    \item Let \textbf{chicken} rest after pressure cooking - it's too hot to handle immediately
    \item Use reserved \textbf{broth} to adjust chicken mixture consistency if needed
    \item Warm tortillas slightly before cutting to prevent cracking
\end{itemize}

\subsection*{Make Ahead \& Storage}
\begin{itemize}
    \item Chicken can be cooked and shredded up to \textit{2~days} ahead
    \item Complete sauce up to \textit{24~hours} in advance
    \item Assembled casserole can be refrigerated up to \textit{24~hours} before baking
    \item If made ahead, add \textit{10~minutes} to baking time
    \item Leftovers keep for \textit{3~days}; reheat covered at \textit{350°F}
\end{itemize}

\subsection*{Serving Suggestions}
\begin{itemize}
    \item Serve with Mexican rice and refried beans
    \item Garnish with diced avocado and fresh cilantro
    \item Offer sour cream and hot sauce on the side
    \item Can be frozen in individual portions for \textit{2~months}
\end{itemize}
