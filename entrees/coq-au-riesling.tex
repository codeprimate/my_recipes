\section*{Ingredients}
\setlength{\columnsep}{20pt}
\begin{multicols}{2}
\noindent
    Chicken thighs, bone-in \dotfill 2\nicefrac{1}{2}-3 lbs \\
    Thick-cut bacon \dotfill 6 oz \\
    Riesling wine \dotfill 2 cups \\
    Leeks \dotfill 4 medium \\
    Cremini mushrooms \dotfill 8 oz \\
    Butter \dotfill \nicefrac{1}{2} cup \\
    Garlic head \dotfill 1 whole \\
    \columnbreak
    Fresh thyme \dotfill 2 sprigs \\
    Fresh parsley \dotfill 2 sprigs \\
    Bay leaves \dotfill 2 \\
    Better than Bouillon, chicken \dotfill 1\nicefrac{1}{2} tsp. \\
    Water \dotfill 1\nicefrac{1}{2} cups \\
    Heavy cream \dotfill \nicefrac{1}{2} cup \\
    Egg yolk \dotfill 1 \\
    All-purpose flour \dotfill 1 Tbsp \\
    Salt and black pepper \dotfill to taste \\
\end{multicols}

\section*{Directions}

\noindent
Cut \textbf{garlic} head in half horizontally; set aside in \textit{Small Bowl~\#1} ---
Clean \textbf{leeks} thoroughly and slice into \nicefrac{1}{2}~inch rounds; set aside in \textit{Medium Bowl~\#1} ---
Quarter \textbf{mushrooms}; set aside in \textit{Medium Bowl~\#2} ---
Dice \textbf{thick-cut bacon} into \nicefrac{1}{4}~inch pieces; set aside in \textit{Small Bowl~\#2} ---
Chop \textbf{parsley} for garnish; set aside in \textit{Small Bowl~\#3} ---
Mix \textbf{Better than Bouillon} with \textbf{hot water} until dissolved; set aside in \textit{Medium Bowl~\#3}

\begin{enumerate}
    \item In \textit{Large Bowl~\#1}, combine \textbf{chicken thighs}, halved \textbf{garlic head} (\textit{Small Bowl~\#1}), 1~sprig \textbf{thyme}, 1~sprig \textbf{parsley}, 1 \textbf{bay leaf}, and 1~tsp. \textbf{salt}. Pour in 2~cups \textbf{Riesling}. Cover and refrigerate for up to \textit{12~hours}.

    \item When ready to cook, remove \textbf{chicken} from marinade and pat dry. Reserve the \textbf{wine marinade} in \textit{Medium Bowl~\#4}. 
    
    \item Melt \nicefrac{1}{4}~of the \textbf{butter} (2~Tbsp.) in a large Dutch oven over medium-high heat. Cook diced \textbf{bacon} (\textit{Small Bowl~\#2}) until golden. Remove with a slotted spoon and transfer to \textit{Medium Bowl~\#5}; set aside.

    \item Add another \nicefrac{1}{4}~of \textbf{butter} (2~Tbsp.), then cook \textbf{leeks} (\textit{Medium Bowl~\#1}) with a pinch of \textbf{salt} until soft, about \textit{5~minutes}. Remove to \textit{Medium Bowl~\#5} with bacon.
    
    \item Add another \nicefrac{1}{4}~of \textbf{butter} (2~Tbsp.) and cook \textbf{mushrooms} (\textit{Medium Bowl~\#2}) until golden. Remove to \textit{Medium Bowl~\#5} with bacon and leeks.

    \item Melt remaining 2~Tbsp. \textbf{butter}. Season \textbf{chicken} with \textbf{salt} and \textbf{pepper}, then brown on all sides until deep golden, about \textit{8-10~minutes}. Transfer to \textit{Large Bowl~\#2}; set aside.

    \item Lower heat and stir 1~Tbsp. \textbf{flour} into pan drippings. Cook for \textit{2~minutes}, then add reserved \textbf{wine marinade} (\textit{Medium Bowl~\#4}) and \textbf{bouillon mixture} (\textit{Medium Bowl~\#3}), scraping bottom of pan.

    \item Return \textbf{chicken} (\textit{Large Bowl~\#2}) to pot. Simmer covered for \textit{90~minutes}, until meat is very tender. Transfer to \textit{Large Bowl~\#2}; set aside.

    \item Whisk \nicefrac{1}{2}~cup \textbf{cream} and \textbf{egg yolk} in \textit{Small Bowl~\#4}. Off heat, whisk into sauce. Return to low heat and cook until thickened.

    \item Add back \textbf{bacon}, \textbf{leeks}, \textbf{mushrooms} (\textit{Medium Bowl~\#5}), and \textbf{chicken} (\textit{Large Bowl~\#2}). Warm through gently. Garnish with chopped \textbf{parsley} (\textit{Small Bowl~\#3}).
\end{enumerate}
