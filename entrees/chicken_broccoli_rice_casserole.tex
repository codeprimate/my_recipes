\section*{Directions}

\noindent
Soften \textbf{cream cheese} at room temp and cube ---
Preheat oven to \textit{375°F} ---
Dice \textbf{chicken} ---
Dice \textbf{onion} ---
Mince \textbf{garlic} ---
Chop \textbf{broccoli} ---
Drain \textbf{corn} and \textbf{mushrooms} ---
Grease a 9x13 inch baking dish with ½~Tbsp. \textbf{butter}

\begin{enumerate}
    \item Cook \textbf{rice} al-dente using 2 cups \textbf{chicken broth} and a \textbf{bay leaf}. Stop your rice cooker 5 minutes early.
    
    \item While rice cooks, bring a pot of salted water to boil. Blanch \textbf{broccoli} for \textit{2~minutes}, then immediately strain in a metal colander rinsing with cold water. Drain well and transfer to a large bowl.
    
    \item In a large skillet, melt 4~Tbsp. \textbf{butter} on medium-low heat. Add \textbf{oil} and \textbf{corn} and increase heat to medium-high to brown and caramelize the corn, about \textit{5~minutes}. Add diced \textbf{onion} and cook until translucent, \textit{5-10~minutes}. Add minced \textbf{garlic} and \textbf{mushrooms}, cooking for another \textit{5~minutes}. Remove from heat.
    
    \item In a large saucepan, melt 2 Tbsp. \textbf{butter}. Add \textbf{rosemary}, \textbf{paprika}, \textbf{MSG}, \textbf{garlic~powder}, \textbf{onion~powder}, \textbf{nutmeg}, \textbf{salt},  \textbf{black~pepper}, and \textbf{red~pepper}. Heat on medium, stirring constantly for \textit{30~seconds}. Sprinkle \textbf{flour} over the butter/spice mixture using a whisk to combine. Cook, stirring constantly for \textit{2~minutes}.
    
    \item Gradually whisk in \textit{2~cups} of \textbf{chicken~broth}, then the \textbf{milk}. Heat and whisk constantly until bubbling and thickened. Add cubes of \textbf{cream~cheese} and cut/whisk until melted and smooth. Remove from heat.
    
    \item In a large bowl, combine and mix blanched \textbf{broccoli}, cooked \textbf{rice}, cooked \textbf{vegetables}, diced \textbf{chicken}, and \textbf{sauce}.
    
    \item Transfer mixture to prepared baking dish and bake at \textit{375°F} for \textit{20~minutes}. Remove from oven to sprinkle liberally with \textbf{crispy fried onions}, and bake another \textit{10~minutes}. Rest \textbf{5-10~minutes} before serving.
\end{enumerate}
