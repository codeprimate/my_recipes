\section*{Ingredients}
\setlength{\columnsep}{20pt}
\begin{multicols}{2}
\noindent
    Bacon strips \dotfill 4-5 \\
    Butter \dotfill 1 Tbsp. \\
    Garlic, jarred minced \dotfill 1-2 Tbsp. \\
    Bush's baked beans \dotfill 1 (28 oz.) can \\
    \columnbreak
    Brown sugar \dotfill 1 Tbsp. \\
    Yellow mustard \dotfill 2 tsp. \\
    Worcestershire sauce \dotfill 2 tsp. \\
    Smoked paprika \dotfill ½ tsp. \\
    Black pepper \dotfill ¼ tsp. \\
\end{multicols}

\section*{Directions}

\noindent
Preheat oven to \textit{375°F} ---
Line a microwave-safe plate with paper towels

\begin{enumerate}
    \item Place \textbf{bacon strips} on the prepared plate and cover with another paper towel. Microwave for \textit{4-6~minutes} until crispy (check at \textit{4~minutes}). Let cool, then crumble.
    
    \item In a stovetop-safe 1.5 to 2-quart baking dish over medium heat, melt \textbf{butter}. Add \textbf{jarred garlic} and sauté for \textit{30-60~seconds} until fragrant.
    
    \item Remove from heat. Add \textbf{baked beans}, \textbf{brown sugar}, \textbf{yellow mustard}, \textbf{Worcestershire sauce}, \textbf{smoked paprika}, and \textbf{black pepper}. Stir to combine thoroughly.
    
    \item Fold in most of the crumbled \textbf{bacon}, reserving some for topping. Sprinkle reserved \textbf{bacon} on top.
    
    \item Bake uncovered for \textit{30-35~minutes} until edges are bubbling and top is slightly caramelized. The sauce should be thickened, not soupy.
    
    \item Let stand for \textit{5~minutes} before serving.
\end{enumerate}
