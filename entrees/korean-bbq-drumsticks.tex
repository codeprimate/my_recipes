\section*{Directions}

\noindent
Mince \textbf{garlic}; set aside in \textit{Small Bowl~\#1} ---
Grate \textbf{ginger}; set aside in \textit{Small Bowl~\#2} ---
Slice \textbf{green onions}; set aside in \textit{Small Bowl~\#3} ---
Pat dry \textbf{drumsticks}

\begin{enumerate}
    \item In \textit{Medium Bowl~\#1}, combine all sauce ingredients: \nicefrac{1}{2}~cup \textbf{gochujang}, 6~Tbsp. \textbf{soy sauce}, 4~Tbsp. \textbf{honey}, 6~Tbsp. \textbf{brown sugar}, \textbf{garlic} (\textit{Small Bowl~\#1}), \textbf{ginger} (\textit{Small Bowl~\#2}), 2~Tbsp. \textbf{sesame oil}, 2~Tbsp. \textbf{rice vinegar}, 4~Tbsp. \textbf{mirin}, and \nicefrac{1}{2}~tsp. \textbf{black pepper}. Mix thoroughly and divide: \nicefrac{2}{3}~for marinade (\textit{Medium Bowl~\#2}), \nicefrac{1}{3}~for basting (\textit{Small Bowl~\#4}, store separately).
    
    \item Place \textbf{drumsticks} in a large container, pour marinade portion (\textit{Medium Bowl~\#2}) over chicken. Cover and refrigerate for \textit{2-4~hours}.
    
    \item Transfer \textbf{drumsticks} to slow cooker, discarding used marinade. Cook on \textit{LOW} for \textit{6-8~hours} or \textit{HIGH} for \textit{3-4~hours} until internal temperature reaches \textit{165°F}.
    
    \item Preheat air fryer to \textit{400°F}. Working in batches of 4-6, brush \textbf{drumsticks} with reserved sauce (\textit{Small Bowl~\#4}) and air fry \textit{3-5~minutes} until crispy, brushing again halfway through. Keep finished batches warm at \textit{200°F}.
    
    \item Garnish with 2~Tbsp. \textbf{sesame seeds} and \textbf{green onions} (\textit{Small Bowl~\#3}). Serve immediately while crispy.
\end{enumerate}
