\section*{Ingredients}

\section*{Directions}

\noindent
Chop \textbf{onions} (4~cups); set aside in \textit{Medium Bowl~\#1} ---
Slice \textbf{mushrooms}; set aside in \textit{Large Bowl~\#1} ---
Chop \textbf{parsley}; set aside in \textit{Small Bowl~\#1}

\begin{enumerate}
    \item Melt 8~Tbsp. \textbf{butter} in a large pot (5+~Qt.) over medium heat.
    \item Add \textbf{onions} (4~cups, \textit{Medium Bowl~\#1}) and cook, stirring until softened, about \textit{5~minutes}.
    \item Add \textbf{mushrooms} (2~lbs., \textit{Large Bowl~\#1}) and sauté for \textit{5~more minutes}.
    \item Stir in 6~cups \textbf{chicken broth}, 2~Tbsp. \textbf{soy sauce}, 1~tsp. \textbf{hot paprika}, 4~tsp. \textbf{sweet paprika}, and 4~tsp. \textbf{dried dill weed}.
    \item Reduce heat to low, cover, and simmer for \textit{15~minutes}.
    \item In \textit{Small Bowl~\#2}, whisk 2~cups \textbf{milk} and 6~Tbsp. \textbf{flour} together; stir into soup until blended.
    \item Cover and simmer for \textit{15~more minutes}, stirring occasionally.
    \item Add 1~cup \textbf{sour cream}, \textbf{parsley} (\nicefrac{1}{2}~cup, \textit{Small Bowl~\#1}), 4~tsp. \textbf{lemon juice}, 1~tsp. \textbf{salt}, \nicefrac{1}{4}~tsp. \textbf{MSG}, and \nicefrac{1}{2}~tsp. \textbf{black pepper}.
    \item Stir over low heat until warmed through, about \textit{3~to~5~minutes}.
    \item Serve immediately.
\end{enumerate}
