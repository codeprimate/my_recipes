\section*{Directions}

\noindent
Preheat oven to \textit{350°F} ---
Bring \textbf{butter} and \textbf{eggs} to room temperature ---
Line baking sheets with parchment or leave ungreased ---
Combine 1\nicefrac{1}{2}~cups \textbf{flour}, 1~tsp. \textbf{baking soda}, 1\nicefrac{1}{2}~tsp. \textbf{cinnamon}, and \nicefrac{1}{2}~tsp. \textbf{salt} in \textit{Medium Bowl~\#1} (dry ingredients)

\begin{enumerate}
    \item In a large bowl, beat 1~cup \textbf{butter}, 1~cup \textbf{brown sugar}, and \nicefrac{1}{2}~cup \textbf{granulated sugar} on medium speed until the mixture is creamy, lightened in color, and no dry sugar remains in the bowl (no loose crystals or gritty patches); about \textit{2--3~minutes}. It may still feel slightly grainy, that is normal and will become smooth when the eggs are added. Scrape bowl.

    \item Add 2 \textbf{eggs} and 1\nicefrac{1}{2}~tsp. \textbf{Mexican vanilla}. Beat well until smooth and fully incorporated, about \textit{1~minute}. Scrape bowl.

    \item Add dry ingredients (\textit{Medium Bowl~\#1}). Mix on low speed until no dry flour remains and dough is uniform, about \textit{30--45~seconds}. Do not overmix.

    \item Add 3~cups \textbf{oats} and 1\nicefrac{1}{2}~cups \textbf{semisweet chocolate chips}. Stir by hand or on low speed until evenly distributed; dough will be thick and chunky.

    \item Cover bowl and chill dough for \textit{15--30~minutes} to reduce spreading during baking.

    \item Drop dough by rounded tablespoonfuls onto ungreased or parchment-lined baking sheets, spacing about \textit{2~inches} apart. Keep portion size consistent for even baking.

    \item Bake at \textit{350°F} for \textit{8--10~minutes} until edges are light golden brown and centers still look soft and slightly underdone. Cookies are done when edges feel set and golden and centers appear moist but not raw; they will set further as they cool. Continue baking in \textit{1~minute} increments if edges are not yet golden.

    \item Cool on baking sheet for \textit{1~minute}, then transfer to a wire rack to cool completely. Repeat with remaining dough.
\end{enumerate}

\begin{itemize}
    \item Large mixing bowl
    \item Hand mixer or stand mixer
    \item Baking sheets (2 recommended)
    \item Parchment paper (optional)
    \item Tablespoon measure or \textit{1\nicefrac{1}{2}--2}~Tbsp. cookie scoop
    \item Measuring cups and spoons

    \item Wire cooling rack
    \item Medium Bowl (dry ingredients)
\end{itemize}

\section*{Hints and Notes}

\subsection*{Yield}
\begin{itemize}
    \item Makes about 4~dozen cookies (depending on scoop size)
\end{itemize}

\subsection*{Mise en Place}
\begin{itemize}
    \item \textit{Medium Bowl~\#1} --- dry ingredients: 1\nicefrac{1}{2}~cups \textbf{flour}, 1~tsp. \textbf{baking soda}, 1\nicefrac{1}{2}~tsp. \textbf{cinnamon}, \nicefrac{1}{2}~tsp. \textbf{salt}
    \item Bring \textbf{butter} and \textbf{eggs} to room temperature \textit{1~hour} before mixing for easier creaming and smoother dough
\end{itemize}

\subsection*{Ingredient Tips}
\begin{itemize}
    \item \textbf{Mexican vanilla} adds warmth and depth; pure vanilla extract can substitute at 1~tsp.
    \item \textbf{Old-fashioned oats} give more texture; \textbf{quick-cooking oats} yield a slightly softer, more uniform cookie.
    \item Use \textbf{semisweet chocolate chips} (not milk) for balance with the sweet dough; \textit{2~cups} for a chunkier cookie if preferred.
\end{itemize}

\subsection*{Preparation Tips}
\begin{itemize}
    \item Room-temperature \textbf{butter} creams properly; cold butter will leave a grainy or stiff dough.
    \item Don't overbake: cookies are done when edges are golden and centers still look soft; they firm up on the rack.
    \item If baking two sheets at once, rotate positions halfway through for even browning.
    \item Dropped cookies spread; \textit{2~inch} spacing prevents merging.
\end{itemize}

\subsection*{Make Ahead \& Storage}
\begin{itemize}
    \item \textbf{Refrigeration:} Dough can be covered and refrigerated up to \textit{2--3~days}. Let sit \textit{10--15~minutes} at room temperature before scooping if very firm. Longer chilling (e.g.\ overnight) reduces spreading further.
    \item \textbf{Freezing:} Portion dough into balls and freeze on a tray, then store in a bag for \textit{2--3~months}. Bake from frozen, adding \textit{1--2~minutes} to bake time.
    \item Baked cookies keep in an airtight container at room temperature for \textit{4--5~days}, or freeze up to \textit{3~months}.
\end{itemize}

\subsection*{Serving Suggestions}
\begin{itemize}
    \item Best at room temperature with cold milk or coffee
    \item Sturdy enough for lunch boxes and cookie tins
    \item Crumble over vanilla ice cream or use for ice cream sandwiches
\end{itemize}
