\section*{Ingredients}

\noindent
    Butter, cool \dotfill 12 Tbsp. (170g) \\
    Powdered sugar \dotfill \nicefrac{2}{3} cup (80g) \\
    Egg yolks, large \dotfill 2 \\
    Mexican vanilla \dotfill 1 tsp. \\
    \columnbreak
    Lemon zest \dotfill 1 Tbsp. \\
    BRM 1:1 G.F. flour \dotfill 1 cup (150g) \\
    Blanched almond flour \dotfill 1 cup (96g) \\
    Cornstarch \dotfill 2 Tbsp. (15g) \\
    Fine salt \dotfill \nicefrac{1}{4} tsp. \\
    Ground cinnamon \dotfill \nicefrac{1}{2} tsp. \\
    Raspberry preserves \dotfill \nicefrac{3}{4} cup
\end{multicols}

\section*{Directions}

\noindent
Zest \textbf{lemon} ---
Measure and whisk together 1~cup \textbf{GF~flour}, 1~cup \textbf{almond~flour}, 2~Tbsp. \textbf{cornstarch}, \nicefrac{1}{4}~tsp. \textbf{salt}, and \nicefrac{1}{2}~tsp. \textbf{cinnamon} in \textit{Medium Bowl~\#1} (dry ingredients) ---
Separate \textbf{eggs}, reserving yolks; combine 2 \textbf{egg~yolks}, 1~tsp. \textbf{vanilla}, and 1~Tbsp. \textbf{lemon~zest} in \textit{Small Bowl~\#1} (wet ingredients) ---
Cut 12~Tbsp. \textbf{butter} into 1-inch cubes ---
Measure \nicefrac{3}{4}~cup \textbf{raspberry~preserves} into \textit{Small Bowl~\#2} (filling) ---
Place \textbf{powdered~sugar} for dusting in \textit{Small Bowl~\#3}

\begin{enumerate}
    \item Ensure \textbf{butter} is cool but pliable (\textit{60--65째F}), it should hold a fingerprint when pressed but not squish through. In a large bowl, beat 12~Tbsp. \textbf{butter} and \nicefrac{2}{3}~cup \textbf{powdered~sugar} with a hand mixer on medium speed until just combined and slightly lightened, about \textit{2--3~minutes}. Scrape bowl.

    \item Add 2 \textbf{egg~yolks}, 1~tsp. \textbf{vanilla}, and 1~Tbsp. \textbf{lemon~zest} (\textit{Small Bowl~\#1}). Beat on medium-low speed until incorporated, about \textit{30~seconds}. Scrape bowl.

    \item Add dry ingredients (\textit{Medium Bowl~\#1}) all at once. Mix on low speed until dough just comes together and looks crumbly with no dry patches, about \textit{45~seconds--1~minute}. Do not overmix.

    \item Turn dough onto a work surface. Gently press together into a cohesive mass, then divide in half. Pat each half into a flat disc about \nicefrac{3}{4}~inch thick. Wrap tightly in plastic wrap.

    \item Refrigerate for a minimum of \textit{4~hours}, preferably \textit{overnight}. This extended rest is critical, it allows the GF starches to fully hydrate, eliminating grittiness and ensuring the dough rolls without cracking.

    \item When ready to bake, remove one disc from the refrigerator. Let rest at room temperature for \textit{5--8~minutes} until pliable but still cold. Preheat oven to \textit{325째F}. Line two baking sheets with parchment.

    \item Roll dough between two sheets of parchment paper to \textit{\nicefrac{1}{4}~inch} thickness. If dough cracks at edges, let it warm slightly; if it becomes sticky, return to refrigerator for \textit{10~minutes}.

    \item Using a 2-inch fluted cutter, cut rounds from dough. Transfer half the rounds (bottoms) to prepared baking sheet. Using a small cutter (\nicefrac{3}{4}~inch), cut a window in the center of remaining rounds (tops). Transfer tops to second baking sheet. Gather scraps, re-roll once, and cut additional cookies.

    \item Place baking sheets in freezer for \textit{15--20~minutes}. This prevents spreading and maintains sharp edges.

    \item Bake cookies directly from freezer at \textit{325째F} for \textit{10--12~minutes} (tops may need \textit{1~minute} less). Cookies are done when edges just begin to turn pale golden---centers will appear slightly underdone but will set upon cooling.

    \item Cool on baking sheet for \textit{5~minutes}, then transfer to wire rack to cool completely, at least \textit{30~minutes}. Repeat with second disc of dough.

    \item To assemble: Place tops on a sheet of parchment. Using a fine-mesh sieve, dust generously with \textbf{powdered~sugar} (\textit{Small Bowl~\#3}). Spread approximately 1~tsp. \textbf{raspberry~preserves} (\nicefrac{3}{4}~cup, \textit{Small Bowl~\#2}) on the flat side of each bottom cookie, leaving a small border. Gently place a sugared top on each, pressing lightly to adhere.
\end{enumerate}

\subsection*{Equipment Required}
\begin{itemize}
    \item Hand mixer
    \item Large bowl for mixing
    \item Two rimmed baking sheets
    \item Parchment paper
    \item Rolling pin
    \item 2-inch fluted round cookie cutter
    \item \nicefrac{3}{4}~inch round or decorative mini cutter
    \item Fine-mesh sieve for dusting
    \item Wire cooling rack
    \item Plastic wrap
    \item Offset spatula (for transferring)
    \item Digital scale (highly recommended)
\end{itemize}

\subsection*{Mise en Place}
\begin{itemize}
    \item Remove \textbf{butter} from refrigerator \textit{15--20~minutes} before mixing, target \textit{60--65째F}
    \item Zest \textbf{lemon} before juicing (zest keeps better on whole fruit)
    \item \textit{Medium Bowl~\#1} --- dry ingredients (1~cup \textbf{GF~flour}, 1~cup \textbf{almond~flour}, 2~Tbsp. \textbf{cornstarch}, \nicefrac{1}{4}~tsp. \textbf{salt}, \nicefrac{1}{2}~tsp. \textbf{cinnamon}) (about 2~cups)
    \item \textit{Small Bowl~\#1} --- wet ingredients (2 \textbf{egg~yolks}, 1~tsp. \textbf{vanilla}, 1~Tbsp. \textbf{lemon~zest})
    \item \textit{Small Bowl~\#2} --- \textbf{raspberry~preserves} (\nicefrac{3}{4}~cup) for assembly
    \item \textit{Small Bowl~\#3} --- \textbf{powdered~sugar} for dusting
    \item Separate \textbf{eggs} while cold (easier separation), then let yolks come to room temperature
    \item Clear freezer space for chilling cut cookies
\end{itemize}

\subsection*{Ingredient Tips}
\begin{itemize}
    \item Use Bob's Red Mill 1:1 in the blue bag, it contains xanthan gum; do not add additional xanthan
    \item Superfine blanched \textbf{almond~flour} prevents grittiness; if yours is coarse, pulse briefly in food processor
    \item European-style \textbf{butter} (82\%+ fat) produces a richer, more tender cookie
    \item For \textbf{preserves}, choose a quality brand (Bonne Maman) with visible fruit pieces; if too loose, simmer with 1~Tbsp. lemon juice until thickened
    \item Fresh \textbf{lemon~zest} is essential, dried zest lacks the volatile oils that brighten the flavor
\end{itemize}

\subsection*{Preparation Tips}
\begin{itemize}
    \item The overnight rest is not optional for GF, shorter chilling produces gritty cookies
    \item Roll between parchment to avoid adding excess flour, which toughens the dough
    \item Work quickly once dough warms; return to refrigerator as needed
    \item Re-roll scraps only once, overworked dough spreads and toughens
    \item Freeze cut cookies before baking to prevent spreading and preserve fluted edges
    \item Bake one sheet at a time in center of oven for even browning
    \item Cookies will feel soft when removed, do not overbake; they firm as they cool
\end{itemize}

\subsection*{Make Ahead \& Storage}
\begin{itemize}
    \item Dough can be refrigerated up to \textit{3~days} or frozen up to \textit{2~months}
    \item Thaw frozen dough overnight in refrigerator before rolling
    \item Unfilled baked cookies keep \textit{5~days} in airtight container at room temperature
    \item Assemble cookies \textit{1--2~days} before serving for best texture, jam softens cookies into ideal tenderness
    \item Filled cookies keep \textit{5--7~days} in airtight container; texture improves on day two
    \item Do not refrigerate filled cookies, condensation softens the powdered sugar
\end{itemize}

\subsection*{Serving Suggestions}
\begin{itemize}
    \item Dust with additional \textbf{powdered~sugar} just before serving if needed
    \item Present on a tiered stand or arranged on parchment-lined box for gifting
    \item Pairs beautifully with strong coffee, Earl Grey tea, or dessert wine
    \item For variation, fill with apricot preserves (traditional Austrian) or lemon curd
    \item Store tops and bottoms separately if making ahead; assemble day of serving for crispest texture
\end{itemize}

\subsection*{Troubleshooting}
\begin{itemize}
    \item \textit{Cookies spread too much}: Dough was too warm; freeze longer before baking, or chill between batches
    \item \textit{Cookies crack when rolling}: Dough is too cold; let rest at room temperature a few more minutes
    \item \textit{Gritty texture}: Insufficient resting time; always chill minimum \textit{4~hours}, preferably overnight
    \item \textit{Cookies too hard}: Overbaked; reduce time by \textit{1--2~minutes}; edges should be barely golden
    \item \textit{Dough too sticky}: Kitchen too warm; work in shorter sessions, returning dough to refrigerator frequently
    \item \textit{Jam leaks out sides}: Too much filling; use scant teaspoon and leave border
\end{itemize}
