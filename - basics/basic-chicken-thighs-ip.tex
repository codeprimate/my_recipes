\section*{Ingredients}
\setlength{\columnsep}{20pt}
\noindent
    Chicken thighs \dotfill 3 lbs \\
    Water \dotfill 2 cups \\
    Better Than Bouillon - Chicken \dotfill 1 Tbsp. \\
    Dried rosemary \dotfill 2 tsp. \\
    Bay leaf \dotfill 1 \\

\section*{Directions}

\noindent
Remove skin from \textbf{thighs} (if bone-in) ---
Trim excess fat ---
Prepare \textbf{stock} with \textbf{Better Than Bouillon}

\begin{enumerate}
    \item Combine \textbf{water} and \textbf{Better Than Bouillon} in a measuring cup, stirring to dissolve completely.
    
    \item Place \textbf{chicken thighs} in the Instant Pot. Add prepared \textbf{stock}, \textbf{dried rosemary}, and \textbf{bay leaf}.
    
    \item Seal lid and set valve to sealing position. Cook on \textit{high} pressure for \textit{15~minutes}.
    
    \item Allow natural release for \textit{5~minutes}, then carefully quick release remaining pressure.
    
    \item Remove \textbf{chicken} and set aside. Strain \textbf{cooking liquid} through fine-mesh strainer, discarding solids. Reserve \textbf{broth} for other recipes.
    
    \item Use \textbf{chicken} as needed - whole, sliced, or shredded.
\end{enumerate}

\subsection*{Equipment Required}
\begin{itemize}
    \item Instant Pot (6-quart or larger)
    \item Measuring cups and spoons
    \item Fine-mesh strainer
    \item Large measuring cup or bowl (for stock preparation)
    \item Storage containers for chicken and broth
    \item Tongs or slotted spoon
    \item Cutting board and knife (for trimming)
\end{itemize}

\subsection*{Mise en Place}
\begin{itemize}
    \item Remove skin from bone-in \textbf{thighs} if necessary
    \item Trim excess fat from \textbf{thighs}
    \item Measure \textbf{water} and \textbf{Better Than Bouillon}
    \item Have storage containers ready for finished products
\end{itemize}

\subsection*{Ingredient Tips}
\begin{itemize}
    \item Both bone-in and boneless \textbf{thighs} work with this timing
    \item Bone-in \textbf{thighs} produce more flavorful \textbf{broth}
    \item Skin can be removed before or after cooking - before is easier
    \item \textbf{Better Than Bouillon} concentration is intentionally strong to account for chicken moisture release
    \item Quality \textbf{dried rosemary} makes a noticeable difference
\end{itemize}

\subsection*{Preparation Tips}
\begin{itemize}
    \item Ensure \textbf{Better Than Bouillon} is fully dissolved before adding to pot
    \item Don't overfill Instant Pot - 3~lbs is appropriate for 6-quart model
    \item If using bone-in \textbf{thighs}, arrange with meatier portions toward bottom for even cooking
    \item \textit{5~minute} natural release prevents vigorous boiling and keeps meat tender
    \item Strain \textbf{broth} while still hot for easier filtering
    \item For shredding, use two forks while \textbf{chicken} is still warm
\end{itemize}

\subsection*{Make Ahead \& Storage}
\begin{itemize}
    \item Cooked \textbf{chicken} keeps refrigerated for \textit{3-4~days}
    \item Reserved \textbf{broth} keeps refrigerated for \textit{4-5~days}
    \item Both \textbf{chicken} and \textbf{broth} freeze well for up to \textit{3~months}
    \item Store \textbf{chicken} and \textbf{broth} separately for maximum flexibility
    \item Label containers with date and contents
    \item Cool to room temperature before refrigerating
    \item Skim fat from \textbf{broth} after refrigeration if desired
\end{itemize}

\subsection*{Serving Suggestions}
\begin{itemize}
    \item Use whole or sliced for grain bowls and salads
    \item Shred for tacos, enchiladas, casseroles, and sandwiches
    \item Dice for soups, fried rice, and pasta dishes
    \item Use reserved \textbf{broth} as soup base or for cooking grains
    \item \textbf{Broth} adds depth to sauces and gravies
    \item Reheat gently to prevent drying - add splash of reserved \textbf{broth} if needed
\end{itemize}
