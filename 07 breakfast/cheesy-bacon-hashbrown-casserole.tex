\section*{Ingredients}
\setlength{\columnsep}{20pt}
\begin{multicols}{2}
\noindent
    Frozen hash browns, thawed \dotfill 30 oz. \\
    Cream of chicken soup \dotfill 10.5 oz. can \\
    Sour cream \dotfill 1\nicefrac{3}{4} cups \\
    Butter, melted \dotfill \nicefrac{1}{2} cup \\
    Onion, diced \dotfill \nicefrac{3}{4} cup \\
    Sharp cheddar cheese \dotfill 2 cups \\
    Gruyère cheese \dotfill 1 cup \\
	\columnbreak
    Bacon, cooked and crumbled \dotfill 1 lb. \\
    Sliced mushrooms \dotfill 8 oz. can \\
    Sun-dried tomatoes \dotfill \nicefrac{1}{2} cup \\
    Garlic powder \dotfill 1 tsp. \\
    Onion powder \dotfill 1 tsp. \\
    Black pepper \dotfill \nicefrac{1}{2} tsp. \\
    Cayenne pepper (optional) \dotfill \nicefrac{1}{4} tsp. \\
    Potato chips, crushed \dotfill 2 cups \\
\end{multicols}

\section*{Directions}

\noindent
Preheat oven to \textit{350°F} ---
Grease a 9~inch×13~inch baking dish ---
Thaw and pat dry \textbf{hash browns}; set aside in \textit{Large Bowl~\#1} ---
Dice \textbf{onion}; set aside in \textit{Small Bowl~\#1} ---
Shred \textbf{cheddar} and \textbf{Gruyère cheese}; combine in \textit{Large Bowl~\#2} ---
Cook and crumble \textbf{bacon}; set aside in \textit{Medium Bowl~\#1} ---
Drain and chop \textbf{mushrooms}; set aside in \textit{Small Bowl~\#2} ---
Crush \textbf{potato chips}; set aside in \textit{Medium Bowl~\#2}

\begin{enumerate}
    \item Rehydrate \textbf{sun-dried tomatoes} in hot water for \textit{10~minutes}, then drain and chop; set aside in \textit{Small Bowl~\#3}.
    \item In \textit{Large Bowl~\#3}, mix \textbf{hash browns} (\textit{Large Bowl~\#1}), \textbf{cream of chicken soup}, \textbf{sour cream}, melted \textbf{butter}, diced \textbf{onion} (\textit{Small Bowl~\#1}), 1\nicefrac{1}{2}~cups \textbf{cheddar and Gruyère cheese} (\textit{Large Bowl~\#2}), \nicefrac{3}{4}~of the crumbled \textbf{bacon} (\textit{Medium Bowl~\#1}), chopped \textbf{mushrooms} (\textit{Small Bowl~\#2}), chopped \textbf{sun-dried tomatoes} (\textit{Small Bowl~\#3}), 1~tsp. \textbf{garlic powder}, 1~tsp. \textbf{onion powder}, \nicefrac{1}{2}~tsp. \textbf{black pepper}, and \nicefrac{1}{4}~tsp. \textbf{cayenne pepper} (if using).
    \item Spread the mixture evenly in the prepared baking dish.
    \item Top with remaining \nicefrac{1}{2}~cup \textbf{cheddar and Gruyère cheese} (\textit{Large Bowl~\#2}) and crushed \textbf{potato chips} (\textit{Medium Bowl~\#2}).
    \item Bake for \textit{50-55~minutes} until golden brown and bubbly.
    \item Remove from oven and sprinkle remaining \nicefrac{1}{4}~of \textbf{bacon} (\textit{Medium Bowl~\#1}) on top.
    \item Let cool for \textit{5-10~minutes} before serving.
\end{enumerate}
