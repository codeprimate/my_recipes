\section*{Ingredients}
\setlength{\columnsep}{20pt}
\begin{multicols}{2}
\noindent
    Bacon and/or breakfast sausage \dotfill 1~lb. \\
    Onion, medium \dotfill 1 \\
    Garlic cloves \dotfill 3--4 \\
    Mushrooms, canned sliced \dotfill 8~oz. can \\
    Frozen hash browns, shredded \dotfill 20~oz. \\
    Butter or reserved fat \dotfill 2~Tbsp. \\
    Eggs, large \dotfill 12 \\
    \columnbreak
    Salt \dotfill 1~tsp. \\
    Black pepper \dotfill \nicefrac{1}{2}~tsp. \\
    White pepper \dotfill \nicefrac{1}{4}~tsp. \\
    Ground mustard \dotfill \nicefrac{1}{2}~tsp. \\
    Thyme, dried \dotfill 1~tsp. \\
    Smoked paprika \dotfill \nicefrac{1}{2}~tsp. \\
    Cheddar cheese, shredded \dotfill 2~cups \\
    Smoked Gouda, shredded \dotfill 1~cup \\
    Cherry tomatoes \dotfill \nicefrac{3}{4}--1~cup \\
    Green onion \dotfill 2--3~stalks \\
\end{multicols}

\section*{Directions}

\noindent
Preheat oven to \textit{350°F} ---
Thaw \textbf{hash browns}; squeeze dry and set aside in \textit{Medium Bowl~\#1} ---
Dice \textbf{onion} and mince \textbf{garlic}; combine in \textit{Small Bowl~\#1} (aromatics) ---
Drain \textbf{mushrooms}; set aside in \textit{Small Bowl~\#2} ---
Shred \textbf{cheddar} and \textbf{smoked Gouda}; combine in \textit{Large Bowl~\#1} ---
Halve \textbf{cherry tomatoes}; set aside in \textit{Small Bowl~\#3} ---
Slice \textbf{green onion}; set aside in \textit{Small Bowl~\#4}

\begin{enumerate}
    \item In a \textit{10--12~inch} deep cast-iron skillet over medium heat, cook \textbf{bacon} and/or \textbf{sausage} until done: \textbf{bacon} is crisp and rendered, \textbf{sausage} has no pink remaining and is crumbled and browned. Transfer to \textit{Medium Bowl~\#2} and crumble; set aside. Pour off all but 2~Tbsp. fat from skillet (or add \textbf{butter} if using only lean sausage). Reserve skillet.
    \item In the same skillet over medium heat, sauté \textbf{onion} and \textbf{garlic} (\textit{Small Bowl~\#1}) until \textbf{onion} is translucent and softened, \textit{about 5--6~minutes}. Add drained \textbf{mushrooms} (\textit{Small Bowl~\#2}) and cook, stirring occasionally, until tender and any liquid has evaporated, \textit{about 4--5~minutes}. Aromatics are done when \textbf{onion} is golden at the edges and \textbf{mushrooms} are lightly browned. Transfer to \textit{Medium Bowl~\#3}; set aside. Wipe skillet if needed and add 1--2~Tbsp. \textbf{butter} or reserved fat.
    \item Spread \textbf{hash browns} (\textit{Medium Bowl~\#1}) in an even layer in the skillet and press down firmly. Cook over medium heat without stirring for \textit{6--8~minutes} until bottom is golden brown and releases easily from the pan. Bottom is done when edges are crisp and the layer holds together when lifted with a spatula. Flip the hash brown layer in one piece (or in two halves if easier) and cook the second side for \textit{5--7~minutes} until golden and crisp. Continue cooking in \textit{1--2~minute} increments if the center is still pale or soft.
    \item While the second side of the \textbf{hash browns} browns, in \textit{Large Bowl~\#2} beat \textbf{eggs} with 1~tsp. \textbf{salt}, \nicefrac{1}{2}~tsp. \textbf{black pepper}, \nicefrac{1}{4}~tsp. \textbf{white pepper}, \nicefrac{1}{2}~tsp. \textbf{ground mustard}, 1~tsp. \textbf{thyme}, and \nicefrac{1}{2}~tsp. \textbf{smoked paprika} until well combined. Stir in cooked \textbf{bacon/sausage} (\textit{Medium Bowl~\#2}), sautéed \textbf{aromatics} (\textit{Medium Bowl~\#3}), and 1\nicefrac{1}{2}~cups of the \textbf{cheddar and Gouda} (\textit{Large Bowl~\#1}).
    \item Pour the \textbf{egg mixture} (\textit{Large Bowl~\#2}) evenly over the \textbf{hash browns} in the skillet. Sprinkle the remaining 1\nicefrac{1}{2}~cups \textbf{cheddar and Gouda} (\textit{Large Bowl~\#1}) over the top and scatter \textbf{cherry tomatoes} (\textit{Small Bowl~\#3}), cut-side up, over the cheese.
    \item Bake at \textit{350°F} for \textit{25--35~minutes} until the center is set: no runny egg when a knife is inserted near the center, top is lightly golden, and the casserole jiggles only slightly when gently shaken. Continue baking in \textit{3--5~minute} increments if the center is still wet.
    \item Let rest \textit{5~minutes}, then sprinkle \textbf{green onion} (\textit{Small Bowl~\#4}) over the top. Serve from the skillet.
\end{enumerate}

\section*{Equipment Required}
\begin{itemize}
    \item 10--12~inch deep cast-iron skillet (oven-safe)
    \item Medium prep bowls (3)
    \item Small prep bowls (4)
    \item Large prep bowls (2)
    \item Whisk
    \item Spatula (for flipping hash browns)
    \item Measuring cups and spoons
    \item Cutting board and chef's knife
\end{itemize}

\section*{Hints and Notes}

\subsection*{Yield}
\begin{itemize}
    \item Serves 6--8 as main breakfast
    \item Makes one \textit{10--12~inch} skillet casserole
\end{itemize}

\subsection*{Mise en Place}
\begin{itemize}
    \item \textit{Medium Bowl~\#1} --- thawed, squeezed-dry \textbf{hash browns} (20~oz.)
    \item \textit{Small Bowl~\#1} --- aromatics: diced \textbf{onion} (1 medium), minced \textbf{garlic} (3--4 cloves)
    \item \textit{Small Bowl~\#2} --- drained \textbf{mushrooms} (8~oz. can)
    \item \textit{Large Bowl~\#1} --- shredded \textbf{cheddar} (2~cups) and \textbf{smoked Gouda} (1~cup) combined; 1\nicefrac{1}{2}~cups go into egg mixture, 1\nicefrac{1}{2}~cups for topping
    \item \textit{Small Bowl~\#3} --- halved \textbf{cherry tomatoes} (\nicefrac{3}{4}--1~cup)
    \item \textit{Small Bowl~\#4} --- sliced \textbf{green onion} (2--3 stalks, for finish)
    \item \textit{Medium Bowl~\#2} --- cooked, crumbled \textbf{bacon/sausage} (created in step 1)
    \item \textit{Medium Bowl~\#3} --- sautéed \textbf{onion}, \textbf{garlic}, and \textbf{mushrooms} (created in step 2)
    \item \textit{Large Bowl~\#2} --- \textbf{egg mixture}: beaten eggs with spices, meat, aromatics, and half the cheese (built in step 4)
    \item Thaw \textbf{hash browns} in refrigerator overnight or under cold running water; squeeze very dry to avoid a soggy crust
\end{itemize}

\subsection*{Ingredient Tips}
\begin{itemize}
    \item Use any combination of \textbf{bacon} and \textbf{sausage} totaling 1~lb.; thick-cut \textbf{bacon} gives the best texture
    \item \textbf{Smoked Gouda} pairs well with the \textbf{smoked paprika}; sharp \textbf{cheddar} adds tang
    \item Frozen \textbf{shredded hash browns} are easiest; squeeze out all excess moisture after thawing
    \item \textbf{White pepper} adds warmth without speckling; omit if you don't have it
    \item \textbf{Cherry tomatoes} can be omitted or reduced if you prefer less moisture on top
\end{itemize}

\subsection*{Preparation Tips}
\begin{itemize}
    \item Flipping the \textbf{hash brown} layer in one piece is easier when the bottom is fully golden and releases cleanly; use a large spatula or flip in two halves
    \item After the meat, leave 2~Tbsp. fat for the aromatics; after sautéing, if the pan is dry, add the full 1--2~Tbsp. \textbf{butter} or reserved fat so \textbf{hash browns} brown and release cleanly
    \item \textbf{Ground mustard} and \textbf{thyme} make the eggs taste much richer; don't skip them
    \item The casserole is done when the center no longer flows and the top is lightly golden; overcooking will dry out the eggs
    \item Rest \textit{5~minutes} before slicing so the egg layer sets for cleaner portions
    \item Best served hot; the hash brown layer is crispiest right out of the oven
\end{itemize}

\subsection*{Make Ahead \& Storage}
\begin{itemize}
    \item Cook \textbf{bacon/sausage} and sauté \textbf{aromatics} up to \textit{1~day} ahead; store covered in refrigerator
    \item Assemble and bake the same day for best texture
    \item Leftovers keep \textit{3--4~days} refrigerated, covered
    \item Reheat portions in a skillet or microwave, or reheat the whole casserole at \textit{350°F} for \textit{15--20~minutes} until hot
    \item Freezes adequately for \textit{2--3~months}; thaw in refrigerator before reheating (\textbf{hash browns} may soften)
\end{itemize}

\subsection*{Serving Suggestions}
\begin{itemize}
    \item Serve hot from the skillet with \textbf{green onion} on top; season with salt and pepper at the table if desired
    \item Hot sauce, sour cream, or avocado on the side
    \item Pair with fruit salad or simple greens
    \item Excellent for brunch or weekend breakfast
\end{itemize}
