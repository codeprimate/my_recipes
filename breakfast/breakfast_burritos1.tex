\section*{Ingredients}
\setlength{\columnsep}{20pt}
\begin{multicols}{2}
\noindent
    Vegetable oil \dotfill 3 Tbsp. \\
    Hash brown potatoes \dotfill 4 cups \\
    Breakfast sausage \dotfill 1 lb. \\
    Eggs \dotfill 8 \\
    \columnbreak
    Chopped green chiles \dotfill 1 can \\
    Black pepper \dotfill \nicefrac{1}{2} tsp. \\
    Salt \dotfill \nicefrac{1}{2} tsp. \\
    Cheddar and/or Jack cheese \dotfill 2 cups \\
    Flour tortillas \dotfill 12 \\
\end{multicols}

\section*{Directions}

\noindent
Preheat large nonstick skillet over medium-high heat ---
Shred \textbf{cheese}; set aside in \textit{Large Bowl~\#1} ---
Beat \textbf{eggs} with \textbf{green chiles}, \nicefrac{1}{2}~tsp. \textbf{salt}, and \nicefrac{1}{2}~tsp. \textbf{black pepper}; set aside in \textit{Medium Bowl~\#1}

\begin{enumerate}
    \item Heat 2~Tbsp. \textbf{oil} in a nonstick skillet. Add \textbf{hash browns}, press down lightly, and cook without moving for \textit{7~minutes}. Drizzle with remaining 1~Tbsp. \textbf{oil}, turn, and cook until browned and hot. Transfer to \textit{Large Bowl~\#2} and set aside.
    \item Cook \textbf{sausage} in the skillet until browned, breaking into crumbles, about \textit{8-10~minutes}. Stir in \textbf{egg} mixture (\textit{Medium Bowl~\#1}) and cook until scrambled, about \textit{5-7~minutes}. Transfer to \textit{Large Bowl~\#3} and set aside.
    \item To assemble burritos, spoon a line of \textbf{hash browns} (\textit{Large Bowl~\#2}) down the center of each \textbf{tortilla}. Top with \textbf{cheese} (\textit{Large Bowl~\#1}) and \textbf{egg} mixture (\textit{Large Bowl~\#3}).
    \item Roll up each \textbf{tortilla} tightly to secure the filling. Serve immediately, or follow freezing instructions.
    \item To freeze, wrap each burrito in foil and store in a freezer bag for up to \textit{1~month}. To reheat, wrap in a damp paper towel and microwave until heated through.
\end{enumerate}
