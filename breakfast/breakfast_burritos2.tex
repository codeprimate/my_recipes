\section*{Ingredients}
\setlength{\columnsep}{20pt}
\begin{multicols}{2}
\noindent
    Breakfast sausage \dotfill 16-24 oz. \\
    Bell pepper \dotfill 1 \\
    Onion \dotfill 1 \\
    Green chiles \dotfill 1 (6 oz.) can \\
    Eggs \dotfill 12 \\
    Milk \dotfill 1\nicefrac{1}{2} cups \\
    \columnbreak
    Sour cream \dotfill \nicefrac{3}{4} cup \\
    Salt \dotfill 1 tsp. \\
    Black pepper \dotfill \nicefrac{1}{2} tsp. \\
    Ground mustard \dotfill \nicefrac{1}{4} tsp. \\
    MSG \dotfill \nicefrac{1}{4} tsp. \\
    Cooking oil (or spray) \dotfill 1 tsp. \\
    Frozen tater tots \dotfill 30 oz. \\
    Tortillas, burrito size \dotfill 12-16 \\
    Shredded cheese \dotfill 3-4 cups
\end{multicols}

\section*{Directions}

\noindent
Preheat oven to \textit{350°F} ---
Dice \textbf{bell pepper} and \textbf{onion}; combine in \textit{Small Bowl~\#1} (aromatics) ---
Drain \textbf{green chiles}; set aside in \textit{Small Bowl~\#2} ---
Shred \textbf{cheese}; set aside in \textit{Large Bowl~\#1}

\begin{enumerate}
    \item Cook the \textbf{breakfast sausage}, \textbf{bell pepper} and \textbf{onion} (\textit{Small Bowl~\#1}), and \textbf{green chiles} (\textit{Small Bowl~\#2}) over medium heat, crumbling the sausage until cooked through, about \textit{10-12~minutes}. Drain grease and transfer to \textit{Large Bowl~\#2}; set aside.
    \item Whisk together the \textbf{eggs}, \textbf{milk}, \textbf{sour cream}, 1~tsp. \textbf{salt}, \nicefrac{1}{2}~tsp. \textbf{black pepper}, \nicefrac{1}{4}~tsp. \textbf{ground mustard}, and \nicefrac{1}{4}~tsp. \textbf{MSG} in \textit{Large Bowl~\#3} until smooth.
    \item Lightly grease a 9~inch×13~inch baking pan with \textbf{oil} or cooking spray. Place the \textbf{tater tots} in the bottom of the pan.
    \item Sprinkle the cooked \textbf{sausage} mixture (\textit{Large Bowl~\#2}) over the \textbf{tater tots} in the prepared pan. Pour the \textbf{egg} mixture (\textit{Large Bowl~\#3}) evenly over the top. 
    \item Bake at \textit{350°F} for \textit{30~minutes} covered, then \textit{30-45~minutes} uncovered until set.
    \item If freezing, let the casserole cool to room temperature. Scoop about \nicefrac{1}{2}~cup of the casserole onto each \textbf{tortilla}, sprinkle with \textbf{cheese} (\textit{Large Bowl~\#1}), fold the sides, and roll up into a burrito.
    \item To freeze, wrap burritos in parchment paper or plastic wrap, and place in a freezer bag. Reheat in the microwave or oven as desired.
\end{enumerate}
