\section*{Directions}

\noindent
Mince \textbf{garlic} and grate \textbf{ginger}; have ready for sauce ---
Slice \textbf{green onions}; set aside in \textit{Small Bowl~\#1} ---
Pat dry \textbf{drumsticks}

\begin{enumerate}
    \item In \textit{Medium Bowl~\#1}, combine all sauce ingredients: \nicefrac{1}{2}~cup \textbf{gochujang}, 6~Tbsp. \textbf{soy sauce}, 4~Tbsp. \textbf{honey}, 6~Tbsp. \textbf{brown sugar}, minced \textbf{garlic}, grated \textbf{ginger}, 2~Tbsp. \textbf{sesame oil}, 2~Tbsp. \textbf{rice vinegar}, 4~Tbsp. \textbf{mirin}, 1~cup \textbf{chicken broth}, and \nicefrac{1}{2}~tsp. \textbf{black pepper}. Mix thoroughly.
    \item Arrange \textbf{drumsticks} in a large Dutch oven. Pour all sauce (\textit{Medium Bowl~\#1}) over \textbf{drumsticks}. Cover with lid and braise in oven at \textit{250°F} for \textit{2~hours} until \textbf{chicken} reaches \textit{165°F} internally and is tender.
    \item Transfer \textbf{drumsticks} to a platter and keep warm. Strain the cooking liquid through a fine-mesh sieve into a saucepan; bring to a simmer over \textit{medium} heat.
    \item In \textit{Small Bowl~\#2}, mix 2~Tbsp. \textbf{water} with 1~Tbsp. \textbf{corn starch} until smooth. Whisk slurry into the simmering liquid and cook, stirring, for \textit{1--2~minutes} until thickened. Sauce is done when it coats the back of a spoon.
    \item Brush \textbf{drumsticks} with the thickened sauce. Air fry at \textit{400°F} for \textit{5~minutes} until skin is crispy, in batches as needed.
    \item Garnish drumsticks with 2~Tbsp. \textbf{sesame seeds} and \textbf{green onions} (\textit{Small Bowl~\#1}). Serve with remaining sauce on the side, \textbf{Sweet~and~Sour~Cabbage}, and \textbf{cooked~white~rice}.
\end{enumerate}

\section*{Equipment Required}
\begin{itemize}
    \item Large Dutch oven (5--7 quart), oven-safe with lid
    \item Medium bowl (for sauce)
    \item Small bowls (for green onions and slurry)
    \item Fine-mesh sieve
    \item Medium saucepan
    \item Whisk
    \item Measuring cups and spoons
    \item Tongs or slotted spoon
    \item Instant-read thermometer
    \item Air fryer
    \item Pastry brush or spoon (for brushing sauce)
\end{itemize}

\section*{Hints and Notes}
\subsection*{Yield}
\begin{itemize}
    \item Serves 4--6
\end{itemize}

\subsection*{Mise en Place}
\begin{itemize}
    \item \textit{Medium Bowl~\#1}: all sauce ingredients combined (minced \textbf{garlic}, grated \textbf{ginger}, gochujang through \textbf{black pepper}, and 1~cup \textbf{chicken broth})
    \item \textit{Small Bowl~\#1}: sliced \textbf{green onions} (garnish)
    \item \textit{Small Bowl~\#2}: corn starch slurry (2~Tbsp. \textbf{water} + 1~Tbsp. \textbf{corn starch}) mixed just before thickening
    \item \textbf{Drumsticks} patted dry; no marinating
\end{itemize}

\subsection*{Ingredient Tips}
\begin{itemize}
    \item \textbf{Gochujang} is Korean fermented chili paste; find it in the Asian aisle or refrigerated section
    \item Low-sodium \textbf{soy sauce} and \textbf{chicken broth} allow better control of salt
    \item Use \textbf{drumsticks} of similar size for even cooking
\end{itemize}

\subsection*{Preparation Tips}
\begin{itemize}
    \item Preheat oven to \textit{250°F} before placing Dutch oven inside
    \item Stir slurry in a small bowl until completely smooth before adding to the saucepan to avoid lumps
    \item If sauce is too thick after adding slurry, thin with a little \textbf{chicken broth} or \textbf{water}
    \item Pat \textbf{drumsticks} dry before brushing with sauce and air frying for better crispness
\end{itemize}

\subsection*{Make Ahead \& Storage}
\begin{itemize}
    \item Refrigerate leftovers in an airtight container for \textit{2--3~days}
    \item Reheat gently on stovetop or in \textit{325°F} oven; sauce may need a splash of \textbf{water} when reheating
\end{itemize}

\subsection*{Serving Suggestions}
\begin{itemize}
    \item Serve over steamed rice or with rice noodles
    \item Pair with \textbf{oi muchim} (cucumber salad) or kimchi
    \item Extra sauce is good drizzled over rice
\end{itemize}
