\section*{Ingredients}
\setlength{\columnsep}{20pt}
\begin{multicols}{2}
\noindent
    Beef chuck roast \dotfill 3-4 lbs \\
    Onion, large \dotfill 1 \\
    Garlic cloves \dotfill 6 \\
    Carrots \dotfill 4 \\
    Potatoes, Yukon Gold \dotfill 4 \\
    Celery stalks \dotfill 2 \\
    Mushrooms \dotfill \nicefrac{1}{2} lb. \\
    \columnbreak

    Tomato paste \dotfill 3 Tbsp. \\
    Dried thyme \dotfill 1 Tbsp. \\
    Dried rosemary \dotfill 1 Tbsp. \\

    Garlic powder \dotfill 1 tsp \\
    Salt \dotfill 2 tsp \\
    Black Pepper, ground \dotfill 1 tsp
\end{multicols}

\section*{Directions}

\noindent
Season the \textbf{beef roast} generously with 2~tsp. \textbf{salt}, 1~tsp. \textbf{pepper}, 1~tsp. \textbf{onion powder}, and 1~tsp. \textbf{garlic powder} ---
Slice the \textbf{onions}; set aside in \textit{Medium Bowl~\#1} ---
Mince the \textbf{garlic}; set aside in \textit{Small Bowl~\#1} ---
Peel and cut the \textbf{carrots} into chunks; set aside in \textit{Medium Bowl~\#2} ---
Cut the \textbf{potatoes} into chunks; set aside in \textit{Medium Bowl~\#3} ---
Chop the \textbf{celery}; set aside in \textit{Small Bowl~\#2} ---
Slice the \textbf{mushrooms}; set aside in \textit{Small Bowl~\#3}

\begin{enumerate}
    \item Heat a large skillet over medium-high heat and sear the \textbf{beef} on all sides until browned. Transfer to \textit{Large Bowl~\#1}; set aside.
    \item Place the sliced \textbf{onions} (\textit{Medium Bowl~\#1}), minced \textbf{garlic} (\textit{Small Bowl~\#1}), \textbf{carrots} (\textit{Medium Bowl~\#2}), \textbf{potatoes} (\textit{Medium Bowl~\#3}), \textbf{celery} (\textit{Small Bowl~\#2}), and \textbf{mushrooms} (\textit{Small Bowl~\#3}) in the crock pot.
    \item In \textit{Medium Bowl~\#4}, whisk together 1~cup \textbf{beef broth}, \nicefrac{1}{2}~cup \textbf{red wine}, 2~Tbsp. \textbf{Worcestershire sauce}, 3~Tbsp. \textbf{tomato paste}, 1~Tbsp. \textbf{dried thyme}, 1~Tbsp. \textbf{dried rosemary}, then add \textbf{bay leaves}.
    \item Pour the mixture (\textit{Medium Bowl~\#4}) over the vegetables in the crock pot.
    \item Place the seared \textbf{beef roast} (\textit{Large Bowl~\#1}) on top of the vegetables.
    \item Cover the crock pot and cook on \textit{low heat for 8-10~hours} or on \textit{high heat for 4-6~hours}, until the beef is tender and easily shreds with a fork.
    \item Once cooked, remove the \textbf{beef} from the crock pot and let it rest for a few minutes before slicing or shredding.
    \item Serve the beef pot roast with the vegetables and drizzle with broth.
\end{enumerate}
