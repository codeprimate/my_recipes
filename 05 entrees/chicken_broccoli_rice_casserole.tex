\section*{Directions}

\noindent
Soften \textbf{cream cheese} at room temp and cube; set aside in \textit{Medium Bowl~\#1} ---
Preheat oven to \textit{375°F} ---
Dice \textbf{chicken}; set aside in \textit{Large Bowl~\#1} ---
Dice \textbf{onion}; set aside in \textit{Small Bowl~\#1} ---
Mince \textbf{garlic}; set aside in \textit{Small Bowl~\#2} ---
Chop \textbf{broccoli}; set aside in \textit{Medium Bowl~\#2} ---
Drain \textbf{corn} and \textbf{mushrooms}; combine in \textit{Medium Bowl~\#3} ---
Grease a 9~inch×13~inch baking dish with \nicefrac{1}{2}~Tbsp. \textbf{butter}

\begin{enumerate}
    \item Cook \textbf{rice} al-dente using 2 cups \textbf{chicken broth} and a \textbf{bay leaf}. Stop your rice cooker 5 minutes early.
    
    \item While rice cooks, bring a pot of salted water to boil. Blanch \textbf{broccoli} for \textit{2~minutes}, then immediately strain in a metal colander rinsing with cold water. Drain well and transfer to a large bowl.
    
    \item In a large skillet, melt 4~Tbsp. \textbf{butter} on medium heat. Add 2~tsp. \textbf{oil} and \textbf{corn} (\textit{Medium Bowl~\#3}) and increase heat to medium-high to brown and caramelize the corn, about \textit{5~minutes}. Add diced \textbf{onion} (\textit{Small Bowl~\#1}) and cook until translucent, \textit{5-10~minutes}. Add minced \textbf{garlic} (\textit{Small Bowl~\#2}) and \textbf{mushrooms} (\textit{Medium Bowl~\#3}), cooking for another \textit{5~minutes}. Remove from heat and transfer to \textit{Large Bowl~\#2}; set aside.
    
    \item In a large saucepan, melt 2~Tbsp. \textbf{butter}. Add 1~tsp. \textbf{rosemary}, 1~tsp. \textbf{paprika}, \nicefrac{1}{4}~tsp. \textbf{MSG}, \nicefrac{1}{2}~tsp. \textbf{garlic powder}, \nicefrac{1}{2}~tsp. \textbf{onion powder}, pinch \textbf{nutmeg}, 1~tsp. \textbf{salt}, \nicefrac{1}{2}~tsp. \textbf{black pepper}, and \nicefrac{1}{4}~tsp. \textbf{red pepper}. Heat on medium, stirring constantly for \textit{30~seconds}. Sprinkle 3~Tbsp. \textbf{flour} over the butter/spice mixture using a whisk to combine. Cook, stirring constantly for \textit{2~minutes}.
    
    \item Gradually whisk in 2~cups of \textbf{chicken broth}, then 2~cups \textbf{milk}. Heat and whisk constantly until bubbling and thickened. Add cubes of \textbf{cream cheese} (\textit{Medium Bowl~\#1}) and cut/whisk until melted and smooth. Remove from heat and transfer to \textit{Large Bowl~\#3}; set aside.
    
    \item In \textit{Large Bowl~\#4}, combine and mix blanched \textbf{broccoli} (\textit{Medium Bowl~\#2}), cooked \textbf{rice}, cooked \textbf{vegetables} (\textit{Large Bowl~\#2}), diced \textbf{chicken} (\textit{Large Bowl~\#1}), and \textbf{sauce} (\textit{Large Bowl~\#3}).
    
    \item Transfer mixture to prepared baking dish and bake at \textit{375°F} for \textit{20~minutes}. Remove from oven to sprinkle liberally with \nicefrac{1}{4}~cup \textbf{crispy fried onions}, and bake another \textit{10~minutes}. Rest \textit{5-10~minutes} before serving.
\end{enumerate}
