\section*{Ingredients}

\begin{multicols}{2}

    All-purpose flour \dotfill 1\nicefrac{1}{2} cups \\

    Baking powder \dotfill 1 tsp. \\
    Cocoa powder \dotfill 1 Tbsp. \\

    Ground ginger \dotfill 2 tsp. \\
    Ground nutmeg \dotfill \nicefrac{1}{2} tsp. \\

    Salt \dotfill \nicefrac{1}{2} tsp. \\
    Vegetable oil \dotfill \nicefrac{2}{3} cup \\
    Applesauce \dotfill \nicefrac{1}{3} cup \\
    Eggs \dotfill 2 \\
    Mexican Vanilla \dotfill 1 tsp. \\
    Powdered sugar \dotfill for dusting \\
\end{multicols}

\section*{Directions}

\noindent
Preheat oven to \textit{350°F}

\begin{enumerate}
    \item In \textit{Medium Bowl~\#1}, whisk together 1~cup \textbf{sugar}, \nicefrac{1}{4}~cup \textbf{brown sugar}, 1\nicefrac{1}{2}~cups \textbf{flour}, 1~tsp. \textbf{baking powder}, \nicefrac{1}{2}~tsp. \textbf{salt}, 1~Tbsp. \textbf{cocoa powder}, 1~Tbsp. \textbf{cinnamon}, 2~tsp. \textbf{ginger}, \nicefrac{1}{2}~tsp. \textbf{nutmeg}, and \nicefrac{1}{4}~tsp. \textbf{allspice}.
    
    \item In \textit{Large Bowl~\#1}, beat \nicefrac{2}{3}~cup \textbf{vegetable oil}, \nicefrac{1}{3}~cup \textbf{applesauce}, \textbf{eggs}, and 1~tsp. \textbf{vanilla extract} until well combined.
    
    \item Add dry ingredients (\textit{Medium Bowl~\#1}) to wet ingredients (\textit{Large Bowl~\#1}) and stir until just blended.
    
    \item Pour batter into a buttered and floured 9~inch×5~inch loaf pan, smoothing the top with a spatula.
    
    \item Bake until golden and a toothpick inserted into the center comes out clean, about \textit{40-45~minutes}.
    
    \item Cool in pan for \textit{10~minutes}, then unmold onto a wire rack to cool completely.
    
    \item Dust with \textbf{powdered sugar} before serving.
\end{enumerate}
