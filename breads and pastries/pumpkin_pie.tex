\section*{Ingredients}
\setlength{\columnsep}{20pt}
\begin{multicols}{2}
\noindent
    Frozen Pie Crusts \dotfill 2 \\
    Cream Cheese \dotfill 8 oz. \\
    Canned Pumpkin \dotfill 2 cups \\
    Sugar \dotfill 1 cup \\
	Salt \dotfill \nicefrac{1}{4} tsp. \\
	Eggs (1 whole plus 2 yolks) \dotfill 3 eggs  \\ 
	\columnbreak
	Heavy cream \dotfill \nicefrac{1}{2} cup \\
	Milk \dotfill \nicefrac{1}{2} cup \\
	Butter, melted \dotfill \nicefrac{1}{4} cup \\
	Vanilla extract \dotfill 1 tsp. \\
	Cinnamon, ground \dotfill \nicefrac{1}{2} tsp. \\
	Ginger, ground \dotfill \nicefrac{1}{2} tsp. \\
	Whipped Cream \dotfill 1 cup \\
\end{multicols}

\section*{Directions}

\noindent
Preheat oven to \textit{350°F} ---
Thaw \textbf{pie crusts} about \textit{15~minutes} at room temperature ---
Lightly beat \textbf{eggs}; set aside in \textit{Small Bowl~\#1}
\subsection*{Pie Crusts}

\begin{enumerate}
	\item Fit a piece of aluminum foil to cover the inside of the shell completely. Fill the shell to the edges with dried beans.
	\item Bake the \textbf{pie crusts} at \textit{350°F} for \textit{10~minutes}, then remove the beans and foil and bake for another \textit{5-10~minutes} until dried and beginning to color. Keep warm.
\end{enumerate}

\subsection*{Filling}
\begin{enumerate}
	\item In \textit{Large Bowl~\#1}, beat the \textbf{cream cheese} with a hand mixer.
	\item Add the \textbf{pumpkin}, then beat until combined.
	\item Add 1~cup \textbf{sugar} and \nicefrac{1}{4}~tsp. \textbf{salt}, then beat until combined.
	\item Add the \textbf{eggs} (\textit{Small Bowl~\#1}), \nicefrac{1}{2}~cup \textbf{milk}, \nicefrac{1}{2}~cup \textbf{cream}, and \nicefrac{1}{4}~cup melted \textbf{butter}, then beat until combined.
	\item Add 1~tsp. \textbf{vanilla}, \nicefrac{1}{2}~tsp. \textbf{cinnamon}, and \nicefrac{1}{2}~tsp. \textbf{ginger}, then beat until combined.
	\item Pour the filling into the pie crusts and loosely wrap the edges with aluminum foil to prevent burning.
	\item Bake for \textit{40~minutes} at \textit{350°F}
	\item Remove foil from edges and bake for \textit{10~minutes}, or until the center is set.
	\item Cool on a wire rack until room temperature. Slice and serve topped with \textbf{whipped cream}.
\end{enumerate}
