\section*{Shortbread Crust}

\noindent
Preheat oven to \textit{350°F} ---
Line 8x8 inch pan with parchment paper, leaving overhang ---
Zest 1 \textbf{lemon} ---
Bring 10~Tbsp. \textbf{butter} to room temperature

\begin{enumerate}
    \item In a medium bowl, whisk together \textbf{gluten free flour}, \textbf{almond flour}, 2~Tbsp. \textbf{cornstarch}, \textbf{powdered sugar}, and ¼~tsp. \textbf{salt}. Add 10~Tbsp. softened \textbf{butter}, \textbf{lemon zest}, and \textbf{vanilla extract}. Mix with a fork or your hands until mixture forms clumps and holds together when pressed.
    
    \item Press dough evenly into prepared pan, creating a uniform ½-inch thick layer. Use the bottom of a measuring cup to compact and smooth the surface. Dock the crust all over with a fork.
    
    \item Bake at \textit{350°F} for \textit{18-22~minutes} until edges are golden brown and center is set. Remove from oven and reduce temperature to \textit{325°F}.
\end{enumerate}

\newpage

\section*{Lemon Curd}

\noindent
Zest 2 \textbf{lemons} ---
Cut 6~Tbsp. \textbf{butter} into tablespoon pieces

\begin{enumerate}
    \item While crust bakes, whisk together \textbf{granulated sugar} and 1~tsp. \textbf{cornstarch} in a medium saucepan. Add \textbf{eggs} and \textbf{egg yolk}, whisking until smooth.
    
    \item Add \textbf{lemon juice}, \textbf{lemon zest}, \textbf{TrueLemon powder}, and ¼~tsp. \textbf{salt}. Whisk to combine.
    
    \item Cook over medium-low heat, stirring constantly with a silicone spatula or wooden spoon, scraping the bottom and sides of the pan. Cook until mixture thickens noticeably and reaches \textit{170°F}, about \textit{8-10~minutes}. The curd should coat the back of a spoon and leave a clear trail when you draw your finger through it.
    
    \item Remove from heat and immediately stir in 6~Tbsp. \textbf{butter}, one piece at a time, until fully incorporated and smooth.
    
    \item Strain the hot curd through a fine-mesh sieve into a bowl or measuring cup to remove \textbf{zest} bits and any cooked egg pieces.
    
    \item Pour strained curd onto the hot pre-baked crust, spreading gently to edges if needed.
    
    \item Bake at \textit{325°F} for \textit{15-18~minutes} until filling is set but still jiggles slightly in the center when gently shaken.
\end{enumerate}

\section*{Finishing}

\begin{enumerate}
    \item Cool completely in the pan on a wire rack for \textit{1~hour}, then refrigerate for at least \textit{3~hours} or overnight.
    
    \item Use parchment overhang to lift bars from pan. Cut into 16 squares using a sharp knife, wiping blade clean between cuts.
    
    \item Just before serving, dust moderately with \textbf{powdered sugar} and sprinkle lightly with \textbf{flaky sea salt}.
\end{enumerate}

\subsection*{Equipment Required}
\begin{itemize}
    \item 8x8 inch baking pan
    \item Parchment paper
    \item Medium mixing bowl (for crust)
    \item Medium saucepan (2-quart capacity)
    \item Fine-mesh sieve or strainer
    \item Silicone spatula or wooden spoon
    \item Whisk
    \item Measuring cups and spoons
    \item Microplane or fine grater (for zest)
    \item Fork (for docking crust)
    \item Instant-read thermometer (optional but helpful)
    \item Sharp knife for cutting
    \item Wire cooling rack
\end{itemize}

\subsection*{Mise en Place}
\begin{itemize}
    \item Bring \textbf{butter} to room temperature for crust (\textit{1~hour})
    \item Zest \textbf{lemons} before juicing (if using fresh)
    \item Cut 6~Tbsp. \textbf{butter} into pieces for filling
    \item Line pan with parchment before starting
    \item Have all ingredients measured and ready
    \item Set up double layer of fine-mesh sieve over a bowl for straining
\end{itemize}

\subsection*{Ingredient Tips}
\begin{itemize}
    \item Bob's Red Mill 1:1 G.F. flour works best; other blends may need adjustment
    \item \textbf{Almond flour} should be finely ground (blanched almond flour, not almond meal)
    \item European-style \textbf{butter} (Kerrygold, Plugrá) has higher fat content and richer flavor
    \item Any shelf-stable \textbf{lemon juice} works; ReaLemon is consistent and reliable
    \item \textbf{TrueLemon powder} significantly amplifies lemon brightness without added liquid
    \item Fresh \textbf{lemon zest} is essential for aromatic oils and complexity
    \item Use the finest grater for \textbf{zest} to avoid bitter white pith
\end{itemize}

\subsection*{Preparation Tips}
\begin{itemize}
    \item Press crust very firmly and evenly; uneven thickness causes uneven baking
    \item Dock crust thoroughly to prevent bubbling during baking
    \item Don't underbake the crust; edges should be golden for structural integrity
    \item Stir curd constantly to prevent curdling; use silicone spatula to scrape sides
    \item Watch for \textit{170°F} on thermometer or coat-the-spoon consistency
    \item Strain while curd is hot for easiest flow through sieve
    \item Pour curd onto hot crust immediately after straining
    \item The filling will continue to set as it cools; slight jiggle is correct
    \item Clean knife between cuts for neat edges
    \item Add \textbf{powdered sugar} and \textbf{salt} just before serving to prevent dissolving
\end{itemize}

\subsection*{Make Ahead \& Storage}
\begin{itemize}
    \item Bars must cool completely and chill at least \textit{3~hours} before cutting
    \item Best made a day ahead; flavors meld and texture improves overnight
    \item Store covered in refrigerator for up to \textit{5~days}
    \item Can be frozen unfrosted for up to \textit{2~months}; thaw in refrigerator overnight
    \item Apply \textbf{powdered sugar} and \textbf{flaky salt} just before serving
    \item For meal prep, cut and store individually wrapped in refrigerator
    \item Bring to cool room temperature before serving for best texture (\textit{15-20~minutes})
\end{itemize}

\subsection*{Serving Suggestions}
\begin{itemize}
    \item Serve as elegant finger food dessert at room temperature
    \item Pair with hot tea, coffee, or sparkling wine
    \item Garnish individual servings with fresh mint or candied lemon peel
    \item Serve alongside fresh berries or whipped cream for plated dessert
    \item The \textbf{flaky salt} provides crucial sweet-savory contrast
    \item Store any leftovers covered in refrigerator; remove \textit{15~minutes} before serving
\end{itemize}
