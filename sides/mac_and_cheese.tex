\section*{Directions}
Preheat oven to \textit{375°F} ---
Lightly grease a 9~inch×13~inch baking dish ---
Shred \textbf{sharp cheddar}, \textbf{Gruyere}, and \textbf{smoked gouda cheeses}; combine in \textit{Large Bowl~\#1} ---
Shred \textbf{Parmesan cheese}; set aside in \textit{Small Bowl~\#1} ---
Chop \textbf{ham}; set aside in \textit{Small Bowl~\#2} ---
Warm \textbf{milk} ---
Combine \textbf{sodium citrate} and 2~Tbsp. water in \textit{Small Bowl~\#3}

\begin{enumerate}
    \item \textbf{Cook the Macaroni:} Bring a large pot of salted water to a rolling boil. Add the \textbf{elbow macaroni} and cook according to package directions until al dente. Drain well and set aside. Do not rinse.
    \item \textbf{Make the Roux:} In a large, heavy-bottomed pot or Dutch oven, melt \nicefrac{1}{2}~cup \textbf{butter} over medium heat. Whisk in \nicefrac{1}{2}~cup \textbf{flour}. Cook, stirring constantly, for \textit{2-3~minutes} until a pale golden roux forms.
    \item \textbf{Build the Cheese Sauce Base:} Gradually whisk in 4~cups \textbf{warmed whole milk}, a little at a time, until smooth. Continue whisking until the sauce thickens about \textit{5-7~minutes}. Stir in \nicefrac{1}{2}~cup \textbf{heavy cream}, 1~tsp. \textbf{smoked paprika}, 1~tsp. \textbf{Hungarian hot paprika}, \nicefrac{1}{2}~tsp. \textbf{garlic powder}, and 1~tsp. \textbf{dry mustard powder}. Season with \nicefrac{1}{4}~tsp. \textbf{salt} and \nicefrac{1}{4}~tsp. \textbf{black pepper} to taste. Stir in \textbf{sodium citrate} (\textit{Small Bowl~\#3}) thoroughly.
    \item \textbf{Melt the Cheeses:} Reduce the heat to low. Add the shredded \textbf{sharp cheddar}, \textbf{Gruyere}, and \textbf{smoked gouda cheeses} (\textit{Large Bowl~\#1}) to the sauce, a handful at a time, stirring constantly until melted and smooth.
    \item \textbf{Combine Macaroni \& Cheese:} Add the cooked macaroni to the cheese sauce and stir gently until fully coated. Fold in the diced \textbf{ham} (\textit{Small Bowl~\#2}).
    \item \textbf{Assemble \& Top:} Transfer the mac and cheese mixture to the prepared baking dish, spreading it evenly. In \textit{Small Bowl~\#4}, combine the \textbf{grated Parmesan cheese} (\nicefrac{1}{2}~cup, \textit{Small Bowl~\#1}) and \nicefrac{1}{2}~cup \textbf{panko breadcrumbs}. Sprinkle this mixture generously over the top.
    \item \textbf{Bake:} Bake for \textit{20-25 minutes}, or until the mac and cheese is bubbly around the edges and the topping is golden brown and crispy.
    \item \textbf{Serve:} Let the mac and cheese stand for \textit{5-10 minutes} before serving. Garnish with fresh \textbf{parsley}.
\end{enumerate}
