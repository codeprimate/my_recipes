\section*{Ingredients}
\setlength{\columnsep}{20pt}
\begin{multicols}{2}
\noindent
    Mushrooms, medium \dotfill 12 (about 1 lb.) \\
    Gluten-free bread \dotfill 3-4 slices \\
    Sweet Italian sausage \dotfill 1 lb. \\
    Shallot, medium \dotfill 1 \\
    Garlic cloves \dotfill 4 \\
    White wine OR chicken broth \dotfill ¼ cup \\
    \columnbreak
    Cream cheese, softened \dotfill 4 oz. \\
    Parmesan, grated \dotfill 4 oz. \\
    Dried oregano \dotfill 1 tsp. \\
    Dried thyme \dotfill ½ tsp. \\
    MSG \dotfill ½ tsp. \\
    Salt \dotfill ¾ tsp. \\
    Black pepper \dotfill ½ tsp. \\
    Red pepper flakes \dotfill ¼ tsp. (optional) \\
    Fresh parsley, chopped \dotfill 2 Tbsp. \\
\end{multicols}

\section*{Directions}

\noindent
Preheat oven to \textit{375°F} ---
Remove \textbf{mushroom} stems and finely chop ---
Clean \textbf{mushroom} caps with damp paper towel ---
Toast \textbf{bread} and process into coarse crumbs ---
Dice \textbf{shallot} ---
Mince \textbf{garlic} ---
Soften \textbf{cream cheese} at room temperature ---
Grate \textbf{parmesan} and divide (3~oz. for filling, 1~oz. for topping) ---
Chop \textbf{parsley}

\begin{enumerate}
    \item Place \textbf{mushroom caps} gill-side up on a rimmed baking sheet. Bake for \textit{10~minutes} to release moisture. Remove and set aside.
    
    \item Toast \textbf{gluten-free bread} until golden and crispy. Process in food processor or crush by hand into coarse crumbs (about 1½~cups). Set aside.
    
    \item In a large skillet, brown \textbf{Italian sausage} over medium heat, breaking into small crumbles, until fully cooked, about \textit{8-10~minutes}. Remove sausage with slotted spoon and set aside. Reserve 2-3~Tbsp. of rendered fat in the skillet, discarding excess.
    
    \item Add diced \textbf{shallot} to the skillet and sauté over medium heat until softened, about \textit{3-4~minutes}.
    
    \item Add minced \textbf{garlic} and chopped \textbf{mushroom stems}. Cook, stirring occasionally, until mushroom liquid has evaporated and mixture is fragrant, about \textit{5-7~minutes}.
    
    \item Add \textbf{breadcrumbs} to the skillet and toast, stirring constantly, for \textit{1-2~minutes} until golden and fragrant.
    
    \item Add \textbf{white wine} or \textbf{chicken broth} and cook, stirring, until liquid is nearly evaporated, about \textit{2-3~minutes}.
    
    \item Remove skillet from heat. Add cooked \textbf{sausage}, softened \textbf{cream cheese}, 3~oz. \textbf{parmesan}, \textbf{oregano}, \textbf{thyme}, \textbf{MSG}, \textbf{salt}, \textbf{black pepper}, and \textbf{red pepper flakes} (if using). Stir until cream cheese is melted and mixture is well combined. Taste and adjust seasoning—the filling should be well-seasoned.
    
    \item Fill each \textbf{mushroom cap} generously with the sausage mixture, mounding slightly. Sprinkle remaining 1~oz. \textbf{parmesan} evenly over the tops.
    
    \item Bake for \textit{20-25~minutes} until \textbf{mushrooms} are tender and tops are golden brown.
    
    \item Garnish with fresh \textbf{parsley} and serve hot.
\end{enumerate}
