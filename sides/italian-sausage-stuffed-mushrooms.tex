\section*{Ingredients}
\setlength{\columnsep}{20pt}
\begin{multicols}{2}
\noindent
    Mushrooms, medium \dotfill 12 (about 1 lb.) \\
    Gluten-free bread \dotfill 3-4 slices \\
    Sweet Italian sausage \dotfill 1 lb. \\
    Shallot, medium \dotfill 1 \\
    Garlic cloves \dotfill 4 \\
    White wine OR chicken broth \dotfill \nicefrac{1}{4} cup \\
    \columnbreak
    Cream cheese, softened \dotfill 4 oz. \\
    Parmesan, grated \dotfill 4 oz. \\
    Dried oregano \dotfill 1 tsp. \\
    Dried thyme \dotfill \nicefrac{1}{2} tsp. \\
    MSG \dotfill \nicefrac{1}{2} tsp. \\
    Salt \dotfill \nicefrac{3}{4} tsp. \\
    Black pepper \dotfill \nicefrac{1}{2} tsp. \\
    Red pepper flakes \dotfill \nicefrac{1}{4} tsp. (optional) \\
    Fresh parsley, chopped \dotfill 2 Tbsp. \\
\end{multicols}

\section*{Directions}

\noindent
Preheat oven to \textit{375°F} ---
Remove \textbf{mushroom} stems and finely chop; set aside in \textit{Small Bowl~\#1} ---
Clean \textbf{mushroom} caps with damp paper towel ---
Toast \textbf{bread} and process into coarse crumbs; set aside in \textit{Medium Bowl~\#1} ---
Dice \textbf{shallot} and mince \textbf{garlic}; combine in \textit{Small Bowl~\#2} (aromatics) ---
Soften \textbf{cream cheese} at room temperature; combine with 3~oz. grated \textbf{parmesan} in \textit{Medium Bowl~\#2} (dairy) ---
Reserve 1~oz. \textbf{parmesan} for topping in \textit{Small Bowl~\#3} ---
Combine 1~tsp. \textbf{oregano}, \nicefrac{1}{2}~tsp. \textbf{thyme}, \nicefrac{1}{2}~tsp. \textbf{MSG}, \nicefrac{3}{4}~tsp. \textbf{salt}, \nicefrac{1}{2}~tsp. \textbf{black pepper}, and \nicefrac{1}{4}~tsp. \textbf{red pepper flakes} in \textit{Small Bowl~\#4} (seasoning) ---
Chop \textbf{parsley}; set aside in \textit{Small Bowl~\#5}

\begin{enumerate}
    \item Place \textbf{mushroom caps} gill-side up on a rimmed baking sheet. Bake for \textit{10~minutes} to release moisture. Remove and set aside.
    
    \item Toast \textbf{gluten-free bread} until golden and crispy. Process in food processor or crush by hand into coarse crumbs (about 1\nicefrac{1}{2}~cups). Transfer to \textit{Medium Bowl~\#1}.
    
    \item In a large skillet, brown \textbf{Italian sausage} over medium heat, breaking into small crumbles, until fully cooked, about \textit{8-10~minutes}. Remove sausage with slotted spoon and transfer to \textit{Medium Bowl~\#3}; set aside. Reserve 2-3~Tbsp. of rendered fat in the skillet, discarding excess.
    
    \item Add \textbf{shallot} and \textbf{garlic} (\textit{Small Bowl~\#2}) to the skillet and sauté over medium heat until softened and fragrant, about \textit{3-4~minutes}.
    
    \item Add chopped \textbf{mushroom stems} (\textit{Small Bowl~\#1}). Cook, stirring occasionally, until mushroom liquid has evaporated, about \textit{5-7~minutes}.
    
    \item Add \textbf{breadcrumbs} (\textit{Medium Bowl~\#1}) to the skillet and toast, stirring constantly, for \textit{1-2~minutes} until golden and fragrant.
    
    \item Add \textbf{white wine} or \textbf{chicken broth} and cook, stirring, until liquid is nearly evaporated, about \textit{2-3~minutes}.
    
    \item Remove skillet from heat. Add cooked \textbf{sausage} (\textit{Medium Bowl~\#3}), \textbf{cream cheese} and 3~oz. \textbf{parmesan} (\textit{Medium Bowl~\#2}), and 1~tsp. \textbf{oregano}, \nicefrac{1}{2}~tsp. \textbf{thyme}, \nicefrac{1}{2}~tsp. \textbf{MSG}, \nicefrac{3}{4}~tsp. \textbf{salt}, \nicefrac{1}{2}~tsp. \textbf{black pepper}, and \nicefrac{1}{4}~tsp. \textbf{red pepper flakes} (\textit{Small Bowl~\#4}). Stir until cream cheese is melted and mixture is well combined. Taste and adjust seasoning—the filling should be well-seasoned.
    
    \item Fill each \textbf{mushroom cap} generously with the sausage mixture, mounding slightly. Sprinkle reserved 1~oz. \textbf{parmesan} (\textit{Small Bowl~\#3}) evenly over the tops.
    
    \item Bake for \textit{20-25~minutes} until \textbf{mushrooms} are tender and tops are golden brown.
    
    \item Garnish with fresh \textbf{parsley} (\textit{Small Bowl~\#5}) and serve hot.
\end{enumerate}

\subsection*{Equipment Required}
\begin{itemize}
    \item Rimmed baking sheet
    \item 12-inch skillet (oven-safe preferred)
    \item Food processor or rolling pin (for breadcrumbs)
    \item Slotted spoon
    \item Small prep bowls (5)
    \item Medium prep bowls (3)
    \item Sharp paring knife
    \item Damp paper towels
    \item Measuring cups and spoons
    \item Mixing spoon or spatula
\end{itemize}

\subsection*{Mise en Place}
\begin{itemize}
    \item Small Bowl \#1 — chopped \textbf{mushroom stems}
    \item Small Bowl \#2 — aromatics: diced \textbf{shallot}, minced \textbf{garlic}
    \item Small Bowl \#3 — 1~oz. \textbf{parmesan} (topping)
    \item Small Bowl \#4 — seasoning: 1~tsp. \textbf{oregano}, \nicefrac{1}{2}~tsp. \textbf{thyme}, \nicefrac{1}{2}~tsp. \textbf{MSG}, \nicefrac{3}{4}~tsp. \textbf{salt}, \nicefrac{1}{2}~tsp. \textbf{black pepper}, \nicefrac{1}{4}~tsp. \textbf{red pepper flakes}
    \item Small Bowl \#5 — chopped \textbf{parsley} (garnish)
    \item Medium Bowl \#1 — \textbf{breadcrumbs} (about 1\nicefrac{1}{2}~cups)
    \item Medium Bowl \#2 — dairy: softened \textbf{cream cheese}, 3~oz. grated \textbf{parmesan}
    \item Medium Bowl \#3 — cooked \textbf{sausage} (set aside after browning, about 2~cups)
    \item Soften \textbf{cream cheese} at room temperature \textit{30~minutes} before cooking
\end{itemize}

\subsection*{Ingredient Tips}
\begin{itemize}
    \item Choose \textbf{mushrooms} with intact caps, 2-2\nicefrac{1}{2} inches in diameter
    \item \textbf{Cremini} or \textbf{baby bella} mushrooms work well for deeper flavor
    \item Use quality \textbf{Italian sausage} with good fat content (not lean)
    \item Fresh \textbf{parmesan} (Parmigiano-Reggiano) preferred over pre-grated
    \item \textbf{White wine}: dry varieties like Pinot Grigio or Sauvignon Blanc work best
    \item Any gluten-free \textbf{bread} works; day-old bread yields better crumbs
\end{itemize}

\subsection*{Preparation Tips}
\begin{itemize}
    \item Pre-baking \textbf{mushroom} caps prevents watery filling
    \item Pat \textbf{mushroom} caps dry after pre-baking
    \item Don't overcrowd \textbf{sausage}—brown in batches if needed
    \item Rendered \textbf{sausage} fat adds flavor; don't discard all of it
    \item Toasting \textbf{breadcrumbs} in the fat adds richness
    \item Filling should be generously seasoned—it will mellow when baked
    \item Mound filling slightly; it will settle during baking
\end{itemize}

\subsection*{Make Ahead \& Storage}
\begin{itemize}
    \item Prep \textbf{mushrooms} and make filling up to \textit{24~hours} ahead
    \item Store filled \textbf{mushrooms} covered in refrigerator
    \item Add extra \textit{5~minutes} baking time if starting from cold
    \item Leftovers keep \textit{3-4~days} refrigerated
    \item Reheat at \textit{350°F} for \textit{10-15~minutes}
    \item Not recommended for freezing (texture suffers)
\end{itemize}

\subsection*{Serving Suggestions}
\begin{itemize}
    \item Serve immediately while cheese is bubbly
    \item Excellent as appetizer (serves 4-6) or side dish (serves 3-4)
    \item Pairs well with roasted meats or grilled steaks
    \item Drizzle with balsamic glaze for presentation
    \item Serve alongside arugula salad with lemon vinaigrette
    \item Can be served room temperature for parties
\end{itemize}
