\section*{Ingredients}
\setlength{\columnsep}{20pt}
\begin{multicols}{2}
\noindent
    Butter \dotfill 1 Tbsp. \\
    Garlic, minced \dotfill 1 Tbsp. \\
    Bush's baked beans \dotfill 1 (28 oz.) can \\
    Brown sugar \dotfill 1 Tbsp. \\
    \columnbreak
    Yellow mustard \dotfill 2 tsp. \\
    Worcestershire sauce \dotfill 2 tsp. \\
    Smoked paprika \dotfill ½ tsp. \\
    Black pepper \dotfill ¼ tsp. \\
    MSG \dotfill pinch
\end{multicols}

\section*{Directions}

\noindent
Preheat oven to \textit{375°F} ---
Mince \textbf{garlic}

\begin{enumerate}
    \item In a small pan or skillet, melt \textbf{butter}. Add \textbf{garlic} and sauté for \textit{30-60~seconds} until fragrant. Remove from heat.
    
    \item In a baking dish, combine \textbf{garlic}, \textbf{baked beans}, \textbf{brown sugar}, \textbf{yellow mustard}, \textbf{Worcestershire sauce}, \textbf{smoked paprika}, and \textbf{black pepper}. Stir thoroughly.
    
    \item Bake at \textit{375°F} uncovered for \textit{30-35~minutes} until edges are bubbling and top is slightly caramelized. The sauce should be thickened, not soupy.
    
    \item Let stand for \textit{5~minutes} before serving.
\end{enumerate}
