\section*{Ingredients}
\setlength{\columnsep}{20pt}
\begin{multicols}{2}
\noindent
    Frozen chopped spinach \dotfill 10 oz. \\
    Artichoke hearts \dotfill 1 (14 oz.) can \\
    Butter \dotfill 2 Tbsp. \\
    Garlic cloves \dotfill 2 \\
    Cream cheese, softened \dotfill 4 oz. \\
    \columnbreak
    Mayonnaise \dotfill ½ cup \\
    Parmesan, grated \dotfill 2 oz. \\
    Mozzarella, shredded \dotfill ¼ cup \\
    Salt \dotfill ¼ tsp. \\
    Black pepper \dotfill ¼ tsp. \\
    Red pepper flakes \dotfill pinch \\
\end{multicols}

\section*{Directions}

\noindent
Preheat oven to \textit{350°F} ---
Thaw \textbf{spinach} and squeeze dry aggressively ---
Drain and rough chop \textbf{artichoke hearts} ---
Mince \textbf{garlic} ---
Soften \textbf{cream cheese} at room temperature

\begin{enumerate}
    \item Melt \textbf{butter} in a large skillet over medium-high heat. Add squeezed \textbf{spinach}, breaking up clumps. Cook for \textit{5-7~minutes}, stirring occasionally, until \textbf{spinach} begins to brown and any remaining moisture evaporates.
    
    \item Add minced \textbf{garlic} and cook for \textit{1~minute} until fragrant. Add chopped \textbf{artichoke hearts} and cook for \textit{2-3~minutes} more. Remove from heat and let cool slightly.
    
    \item In a large bowl, combine softened \textbf{cream cheese}, \textbf{mayonnaise}, \textbf{Parmesan}, \textbf{salt}, \textbf{black pepper}, and \textbf{red pepper flakes} until smooth.
    
    \item Fold in the cooled \textbf{spinach} and \textbf{artichoke} mixture until evenly distributed.
    
    \item Transfer to a 1-quart baking dish and top with shredded \textbf{mozzarella}.
    
    \item Bake for \textit{25-30~minutes} until bubbly and golden brown on top.
    
    \item Let rest for \textit{5-10~minutes} before serving.
\end{enumerate}

\subsection*{Equipment Required}
\begin{itemize}
    \item 1-quart baking dish (ceramic or glass)
    \item Large skillet or sauté pan
    \item Large mixing bowl
    \item Cutting board and sharp knife
    \item Measuring cups and spoons
    \item Rubber spatula or wooden spoon
    \item Kitchen towel or cheesecloth (for squeezing spinach)
    \item Colander (for draining artichokes)
\end{itemize}

\subsection*{Mise en Place}
\begin{itemize}
    \item Thaw \textbf{frozen spinach} overnight in refrigerator or use microwave defrost
    \item Remove \textbf{cream cheese} from refrigerator at least \textit{1~hour} before starting
    \item Squeeze \textbf{spinach} as dry as possible - this is critical for texture
    \item Drain \textbf{artichoke hearts} thoroughly and chop before cooking
    \item Have all ingredients measured and ready before starting
\end{itemize}

\subsection*{Ingredient Tips}
\begin{itemize}
    \item Use artichoke hearts packed in water rather than marinated varieties
    \item Frozen chopped spinach works better than whole leaf for this recipe
    \item The key to success is aggressively squeezing the \textbf{spinach} - wrap in kitchen towel and twist hard
    \item Full-fat dairy products yield the best texture and flavor
    \item Freshly grated \textbf{Parmesan} provides better flavor than pre-grated
\end{itemize}

\subsection*{Preparation Tips}
\begin{itemize}
    \item Don't skip the browning step - caramelized \textbf{spinach} adds deep, nutty flavor
    \item Watch for golden bits forming on the \textbf{spinach} - that's where the flavor develops
    \item Let vegetables cool slightly before mixing with dairy to prevent separation
    \item Rough chop \textbf{artichokes} for better texture - avoid fine chopping
    \item If top browns too quickly during baking, cover loosely with foil
\end{itemize}

\subsection*{Make Ahead \& Storage}
\begin{itemize}
    \item Can be assembled up to \textit{24~hours} in advance and refrigerated
    \item If made ahead, bring to room temperature for \textit{30~minutes} before baking
    \item This recipe reheats exceptionally well without separation
    \item Leftovers can be refrigerated for up to \textit{4~days}
    \item Reheat in \textit{325°F} oven until warm, about \textit{15-20~minutes}, or microwave and stir
\end{itemize}

\subsection*{Serving Suggestions}
\begin{itemize}
    \item Serve with toasted baguette slices, pita chips, or crackers
    \item Fresh vegetables like bell peppers, carrots, and celery make excellent dippers
    \item For best texture, allow to cool slightly before serving
    \item Garnish with additional \textbf{Parmesan} or fresh cracked pepper if desired
    \item Pairs well with a crisp white wine, light beer, or sparkling water
\end{itemize}
