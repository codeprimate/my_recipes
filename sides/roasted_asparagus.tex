\section*{Ingredients}

\begin{multicols}{2}
\noindent

	\columnbreak
    Butter \dotfill 2\nicefrac{1}{2} Tbsp. \\

    Dried minced onion \dotfill \nicefrac{1}{2} tsp. \\
    Chicken broth \dotfill 3 Tbsp. \\
    Cornstarch \dotfill 1 Tbsp. \\
    Water \dotfill 2 Tbsp. \\
\end{multicols}

\section*{Roasted Asparagus}

\noindent
Preheat oven to \textit{400Â°F} ---
Trim woody ends from \textbf{asparagus}; set aside in \textit{Medium Bowl~\#1} ---
Mince \textbf{garlic}; set aside in \textit{Small Bowl~\#1} ---
Zest and juice \textbf{lemon} to yield 1\nicefrac{1}{2}~tsp. zest and 2~Tbsp. juice; set zest in \textit{Small Bowl~\#2} and juice in \textit{Small Bowl~\#3} ---
Mix 1~Tbsp. \textbf{cornstarch} with 2~Tbsp. cold \textbf{water} to make slurry in \textit{Small Bowl~\#4}

\begin{enumerate}
    \item Place trimmed \textbf{asparagus} (\textit{Medium Bowl~\#1}) on a baking sheet. Drizzle with 1\nicefrac{1}{2}~Tbsp. \textbf{olive oil} and sprinkle with \nicefrac{1}{2}~tsp. \textbf{dried thyme}, \textbf{lemon zest} (1\nicefrac{1}{2}~tsp., \textit{Small Bowl~\#2}), \nicefrac{1}{2}~tsp. \textbf{salt}, and \nicefrac{1}{4}~tsp. \textbf{black pepper}. Toss to coat evenly.
    
    \item Arrange \textbf{asparagus} in a single layer and roast for \textit{30-40~minutes} until tender-crisp and lightly browned.
\end{enumerate}

\section*{Lemon-Garlic Butter Sauce}

\begin{enumerate}
    \item While \textbf{asparagus} roasts, melt 2\nicefrac{1}{2}~Tbsp. \textbf{butter} in a small saucepan over medium-low heat.
    \item Add minced \textbf{garlic} (\textit{Small Bowl~\#1}), 1~tsp. \textbf{dried parsley}, and \nicefrac{1}{2}~tsp. \textbf{dried minced onion}.
    \item Cook for \textit{1-2~minutes} until garlic is fragrant.
    \item Add 3~Tbsp. \textbf{chicken broth} and \textbf{lemon juice} (2~Tbsp., \textit{Small Bowl~\#3}). Bring to a gentle simmer.
    \item While whisking constantly, slowly pour the cornstarch slurry (\textit{Small Bowl~\#4}) into the simmering sauce.
    \item Continue whisking and simmer for \textit{2-3~minutes} until sauce is thickened and no longer cloudy.
    \item Season with additional \textbf{salt} to taste. Remove from heat.
    \item Pour warm sauce over roasted \textbf{asparagus} and serve immediately.
\end{enumerate}
