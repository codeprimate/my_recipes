\section*{Directions}

\noindent
Chop \textbf{onion}, \textbf{carrots}, \textbf{celery}, \textbf{potatoes}, and \textbf{zucchini}; combine in \textit{Large Bowl~\#1} ---
Mince \textbf{garlic}; set aside in \textit{Small Bowl~\#1} ---
Drain \textbf{mushrooms}; set aside in \textit{Medium Bowl~\#1}

\begin{enumerate}
    \item Brown 1~lb. \textbf{sweet Italian sausage} and 1~lb. \textbf{hot Italian sausage} into small chunks. Transfer to \textit{Medium Bowl~\#2}; set aside.
    \item Combine all ingredients in a large stock pot: browned \textbf{sausage} (\textit{Medium Bowl~\#2}), \textbf{onion}, \textbf{carrots}, \textbf{celery}, \textbf{potatoes}, and \textbf{zucchini} (\textit{Large Bowl~\#1}), \textbf{garlic} (\textit{Small Bowl~\#1}), 1~lb. \textbf{frozen vegetables}, \textbf{mushrooms} (\textit{Medium Bowl~\#1}), 12~oz. \textbf{tomato sauce}, 24~oz. \textbf{diced tomatoes}, 12-14~cups \textbf{chicken broth}, 1~tsp. \textbf{Italian herb mix}, 1~tsp. \textbf{basil}, 3~Tbsp. \textbf{parsley}, 3~ \textbf{bay leaves}, \nicefrac{1}{2}~tsp. \textbf{oregano}, 1~Tbsp. \textbf{sugar}, 1~tsp. \textbf{onion salt}, and 1~tsp. \textbf{MSG} (optional).
    \item Bring to a boil, reduce heat and simmer covered for \textit{30~minutes}.
    \item Stir occasionally and vigorously to thicken broth.
    \item For canning: process at \textit{15~pounds} for \textit{90~minutes}.
\end{enumerate}

\noindent
\textit{Add 1-2 cups of water per quart when reheating.}
