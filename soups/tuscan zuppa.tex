\section*{Ingredients}
\setlength{\columnsep}{20pt}
\begin{multicols}{2}
\noindent
    Italian sausage \dotfill 1 lb. \\
    Onion, medium \dotfill 1 \\
    Garlic \dotfill 4-6 cloves \\
    White wine (optional) \dotfill \nicefrac{1}{4} cup \\
    Kale \dotfill 1 bunch \\
    Potatoes, large \dotfill 3-4 \\
    \columnbreak
    Chicken broth \dotfill 8 cups \\
    Cannellini beans \dotfill 1 (15 oz.) can \\
    Heavy cream \dotfill 1 cup \\
    Salt \dotfill \nicefrac{1}{2} tsp.\\
    Pepper \dotfill \nicefrac{1}{4} tsp. \\
	Red pepper flakes \dotfill \nicefrac{1}{4} tsp. \\
    MSG (optional) \dotfill \nicefrac{1}{4} tsp.
\end{multicols}

\section*{Directions}

\noindent
Dice \textbf{onion}; set aside in \textit{Small Bowl~\#1} ---
Mince \textbf{garlic}; set aside in \textit{Small Bowl~\#2} ---
Dice \textbf{potatoes} into \nicefrac{1}{4}~inch pieces; set aside in \textit{Medium Bowl~\#1} ---
Remove stems from \textbf{kale} and tear leaves into bite sized pieces; set aside in \textit{Medium Bowl~\#2}

\begin{enumerate}
    \item Brown 1~lb. \textbf{Italian sausage} over medium heat until fully cooked. Remove and transfer to \textit{Medium Bowl~\#3}; set aside, retaining drippings in the pot.
    \item Sauté \textbf{onions} (\textit{Small Bowl~\#1}) in the same pot until translucent, then add \textbf{garlic} (\textit{Small Bowl~\#2}) and cook for an additional \textit{1~minute}.
    \item Deglaze the pan with \nicefrac{1}{4}~cup \textbf{white wine} (optional).
    \item Return \textbf{sausage} (\textit{Medium Bowl~\#3}) to the pot along with 8~cups \textbf{chicken broth} and \textbf{potatoes} (\textit{Medium Bowl~\#1}). Bring to a boil, then simmer until potatoes are tender, stirring occasionally, about \textit{10-15~minutes}.
    \item Stir in \textbf{kale} (\textit{Medium Bowl~\#2}) and \textbf{cannellini beans} (do not drain). Simmer until kale is wilted, about \textit{5~minutes}.
    \item Stir in 1~cup \textbf{heavy cream}. Season with \nicefrac{1}{2}~tsp. \textbf{salt}, \nicefrac{1}{4}~tsp. \textbf{pepper}, \nicefrac{1}{4}~tsp. \textbf{red pepper flakes}, and \nicefrac{1}{4}~tsp. \textbf{MSG} (optional) to taste.
\end{enumerate}
