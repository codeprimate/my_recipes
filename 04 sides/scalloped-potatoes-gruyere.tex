\section*{Ingredients}
\setlength{\columnsep}{20pt}
\begin{multicols}{2}
\noindent
    Yukon Gold potatoes \dotfill 3~lb. \\
    Unsalted butter \dotfill 3~Tbsp. \\
    Garlic cloves \dotfill 2 \\
    All-purpose flour \dotfill 3~Tbsp. \\
    Whole milk \dotfill 1~cup \\
    \columnbreak
    Heavy cream \dotfill 2~cups \\
    Kosher salt \dotfill 1\nicefrac{1}{2}~tsp. \\
    Black pepper \dotfill \nicefrac{1}{2}~tsp. \\
    Nutmeg \dotfill pinch \\
    Dried thyme \dotfill \nicefrac{3}{4}~tsp. \\
    Gruyère cheese \dotfill 6~oz. \\
    Parmesan cheese \dotfill 3~oz. (\nicefrac{3}{4}~cup grated) \\
\end{multicols}

\section*{Directions}

\noindent
Preheat oven to \textit{375°F} ---
Butter a standard 9~inch×13~inch (3~quart) baking dish ---
Mince \textbf{garlic}; set aside in \textit{Small Bowl~\#1} ---
Grate \textbf{Parmesan}; reserve \nicefrac{1}{4}~cup for topping in \textit{Small Bowl~\#2} ---
Shred \textbf{Gruyère}; combine with remaining \nicefrac{1}{2}~cup \textbf{Parmesan} in \textit{Medium Bowl~\#1} (cheese for sauce) ---
Combine \textbf{salt}, \textbf{pepper}, \textbf{nutmeg}, and \textbf{dried thyme} in \textit{Small Bowl~\#3} (seasoning) ---
Slice \textbf{potatoes} \nicefrac{1}{4}~inch thick; hold in cold water until layering, then pat dry

\begin{enumerate}
    \item Make the cream sauce in a large saucepan over medium heat: Melt 3~Tbsp. \textbf{butter}. Add minced \textbf{garlic} (\textit{Small Bowl~\#1}) and cook for \textit{30~seconds} until fragrant. Whisk in 3~Tbsp. \textbf{flour} and cook, stirring, for \textit{1--2~minutes} until pale golden and no raw flour smell remains.

    \item Gradually whisk in 1~cup \textbf{milk} and 2~cups \textbf{heavy cream}. Bring to a gentle simmer, stirring often. Cook for \textit{3--5~minutes} until thickened enough to coat the back of a spoon. Remove from heat. Stir in \textbf{seasoning} (\textit{Small Bowl~\#3}) and \textbf{Gruyère} with \nicefrac{1}{2}~cup \textbf{Parmesan} (\textit{Medium Bowl~\#1}) until cheese melts and sauce is smooth. Taste and adjust \textbf{salt} and \textbf{pepper}.

    \item Pat \textbf{potato} slices dry with towels. Layer \textbf{potatoes} and sauce in the prepared dish: Arrange one-third of the \textbf{potato} slices in an overlapping layer. Spoon one-third of the sauce over the potatoes. Repeat twice more, ending with sauce. Spread sauce evenly so potatoes are well coated. Sprinkle reserved \nicefrac{1}{4}~cup \textbf{Parmesan} (\textit{Small Bowl~\#2}) over the top.

    \item Cover tightly with foil. Bake at \textit{375°F} for \textit{40--45~minutes} until \textbf{potatoes} are tender when pierced with a knife and sauce is bubbling at the edges. Potatoes should offer no resistance; center may still jiggle slightly. Continue baking in \textit{5~minute} increments if center \textbf{potatoes} are still firm.

    \item Remove foil. Bake uncovered for \textit{18--22~minutes} until top is golden brown and bubbly and sauce has thickened. The dish is done when a knife pierces the center \textbf{potatoes} easily and the top has visible browning. Continue baking in \textit{2--3~minute} increments if top needs more color.

    \item Let rest for \textit{15--20~minutes} before serving. Resting allows the sauce to set and slices to hold together when scooped. Serve warm.
\end{enumerate}

\subsection*{Yield}
\begin{itemize}
    \item Serves 5--6
\end{itemize}

\subsection*{Equipment Required}
\begin{itemize}
    \item Standard 9~inch×13~inch (3~quart) baking dish
    \item Mandoline or sharp chef's knife (for uniform \nicefrac{1}{4}~inch slices)
    \item Large saucepan (3--4~quart)
    \item Whisk
    \item Small prep bowls (3), Medium prep bowl (1)
    \item Aluminum foil
    \item Measuring cups and spoons
\end{itemize}

\subsection*{Mise en Place}
\begin{itemize}
    \item Small Bowl \#1 --- minced \textbf{garlic}
    \item Grate \textbf{Parmesan}; divide into \nicefrac{1}{4}~cup (topping) and \nicefrac{1}{2}~cup (for sauce)
    \item Small Bowl \#2 --- \nicefrac{1}{4}~cup \textbf{Parmesan} (topping)
    \item Small Bowl \#3 --- seasoning: \textbf{salt}, \textbf{pepper}, \textbf{nutmeg}, \textbf{dried thyme}
    \item Medium Bowl \#1 --- \textbf{Gruyère} and \nicefrac{1}{2}~cup \textbf{Parmesan} (for sauce)
    \item Slice \textbf{potatoes} just before layering to minimize browning
\end{itemize}

\subsection*{Ingredient Tips}
\begin{itemize}
    \item \textbf{Yukon Gold}: Creamier than russet, holds shape well; russet works but may be slightly mealier
    \item \textbf{Gruyère}: Shred from a block for smooth melting; pre-shredded can be grainy
    \item \textbf{Parmesan}: Freshly grated preferred for topping
    \item \textbf{Dried thyme}: \nicefrac{1}{2}~tsp. if you prefer a lighter herb note
\end{itemize}

\subsection*{Preparation Tips}
\begin{itemize}
    \item Uniform \textbf{potato} slices ensure even cooking; mandoline gives best results
    \item Do not rinse \textbf{potato} slices---starch released during baking helps thicken the sauce
    \item Sauce should coat the back of a spoon before layering; it will thin as \textbf{potatoes} release moisture, then thicken again during baking and resting
    \item Resting \textit{15--20~minutes} is essential---sauce sets and slices hold together; serving too soon yields soupy portions
    \item If top browns too quickly, tent loosely with foil during the uncovered bake
\end{itemize}

\subsection*{Make Ahead \& Storage}
\begin{itemize}
    \item Assemble up to \textit{24~hours} ahead: layer \textbf{potatoes} and sauce, cover tightly, refrigerate. Add \textit{5--10~minutes} to covered bake time if starting from cold.
    \item Leftovers keep \textit{3--4~days} refrigerated
    \item Reheat at \textit{350°F} for \textit{15--20~minutes} until hot; add a splash of \textbf{cream} or \textbf{milk} if sauce seems dry
    \item Freezing not recommended---texture suffers
\end{itemize}

\subsection*{Serving Suggestions}
\begin{itemize}
    \item Pairs well with pan-seared salmon, roasted chicken, or beef
    \item Serve while still warm; cheese is at its best when freshly rested
    \item A simple green salad or roasted asparagus balances the richness
\end{itemize}
