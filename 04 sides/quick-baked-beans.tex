\section*{Ingredients}
\setlength{\columnsep}{20pt}
\begin{multicols}{2}
\noindent
    Butter \dotfill 1 Tbsp. \\
    Garlic, minced \dotfill 1 Tbsp. \\
    Bush's baked beans \dotfill 1 (28 oz.) can \\
    Brown sugar \dotfill 1 Tbsp. \\
    \columnbreak
    Yellow mustard \dotfill 2 tsp. \\
    Worcestershire sauce \dotfill 2 tsp. \\
    Smoked paprika \dotfill \nicefrac{1}{2} tsp. \\
    Black pepper \dotfill \nicefrac{1}{4} tsp. \\
    MSG \dotfill pinch
\end{multicols}

\section*{Directions}

\noindent
Preheat oven to \textit{375°F} ---
Mince \textbf{garlic}; set aside in \textit{Small Bowl~\#1}

\begin{enumerate}
    \item In a small pan or skillet, melt 1~Tbsp. \textbf{butter}. Add \textbf{garlic} (1~Tbsp., \textit{Small Bowl~\#1}) and sauté for \textit{30-60~seconds} until fragrant. Remove from heat and transfer to \textit{Small Bowl~\#2}; set aside.
    
    \item In a baking dish, combine \textbf{garlic} (\textit{Small Bowl~\#2}), \textbf{baked beans}, 1~Tbsp. \textbf{brown sugar}, 2~tsp. \textbf{yellow mustard}, 2~tsp. \textbf{Worcestershire sauce}, \nicefrac{1}{2}~tsp. \textbf{smoked paprika}, and \nicefrac{1}{4}~tsp. \textbf{black pepper}. Stir thoroughly.
    
    \item Bake at \textit{375°F} uncovered for \textit{30-35~minutes} until edges are bubbling and top is slightly caramelized. The sauce should be thickened, not soupy.
    
    \item Let stand for \textit{5~minutes} before serving.
\end{enumerate}
