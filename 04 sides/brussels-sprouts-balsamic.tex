\section*{Ingredients}
\setlength{\columnsep}{20pt}
\begin{multicols}{2}
\noindent
    Brussels sprouts \dotfill 1\nicefrac{1}{2} lbs. \\
    High-heat oil \dotfill 2 Tbsp. \\
    Kosher salt \dotfill \nicefrac{3}{4} tsp. \\
    \columnbreak    
    Black pepper \dotfill \nicefrac{1}{2} tsp. \\
    Unsalted butter \dotfill 6 Tbsp. \\
    Fresh rosemary sprigs \dotfill 2-3 \\
    Balsamic glaze \dotfill 3 Tbsp. \\
    Flaky finishing salt \dotfill to taste \\
\end{multicols}

\section*{Directions}

\noindent
Preheat air fryer to \textit{400°F} ---
Trim and halve \textbf{brussels sprouts}; set aside in \textit{Large Bowl~\#1} ---
Have \textbf{balsamic glaze} ready for finishing

\begin{enumerate}
    \item In \textit{Large Bowl~\#2}, toss halved \textbf{brussels sprouts} (\textit{Large Bowl~\#1}) with 2~Tbsp. \textbf{high-heat oil}, \nicefrac{3}{4}~tsp. \textbf{kosher salt}, and \nicefrac{1}{2}~tsp. \textbf{black pepper} until evenly coated.
    
    \item Arrange \textbf{brussels~sprouts} in a single layer in the air fryer basket, cut side down when possible. Work in batches if necessary to avoid overcrowding.
    
    \item Air fry at \textit{400°F} for \textit{12-14~minutes}, shaking the basket halfway through, until outer leaves are charred and crispy and cut surfaces are deeply caramelized with dark brown to black edges.
    
    \item While \textbf{brussels~sprouts} are cooking, melt \textbf{butter} in a small saucepan (preferably light-colored to monitor browning) over medium heat.
    
    \item Continue cooking, swirling occasionally, for \textit{3-5~minutes} after the \textbf{butter} foams. Watch carefully as milk solids turn golden brown and develop a nutty aroma. Remove from heat immediately when the solids reach a deep amber color.
    
    \item Add \textbf{rosemary~sprigs} to the brown \textbf{butter} off heat. Let steep for \textit{2-3~minutes} while the \textbf{sprouts} finish cooking.
    
    \item Remove \textbf{rosemary~sprigs} and whisk in \textbf{balsamic~glaze} until loosely emulsified. The mixture will remain somewhat separated---this is expected.
    
    \item Transfer hot \textbf{brussels~sprouts} directly from air fryer to a large serving bowl.
    
    \item Immediately drizzle balsamic brown \textbf{butter} over the \textbf{sprouts} and toss gently to coat evenly. The residual heat will help distribute the sauce.
    
    \item Taste and adjust seasoning. Finish with \textbf{flaky~salt} to taste. Serve immediately while \textbf{sprouts} are still crispy and the \textbf{butter} is warm.
\end{enumerate}

\subsection*{Equipment Required}
\begin{itemize}
    \item Air fryer (basket style preferred)
    \item Large mixing bowl
    \item Small saucepan (light-colored interior for monitoring butter)
    \item Whisk
    \item Large serving bowl
    \item Measuring cups and spoons
    \item Sharp knife and cutting board
\end{itemize}

\subsection*{Mise en Place}
\begin{itemize}
    \item Trim and halve all \textbf{brussels~sprouts} before starting
    \item Measure \textbf{balsamic~glaze} and have ready
    \item Have \textbf{butter} at room temperature for easier melting
    \item Strip \textbf{rosemary} sprigs if using leaves from a larger bunch
    \item Start brown \textbf{butter} when \textbf{sprouts} have \textit{5-6~minutes} remaining for optimal temperature synchronization
\end{itemize}

\subsection*{Ingredient Tips}
\begin{itemize}
    \item Use high-heat oil with smoke point above \textit{400°F}: avocado oil (\textit{520°F}) or refined coconut oil (\textit{450°F}) are ideal
    \item Avoid extra virgin olive oil as it will smoke
    \item Choose \textbf{balsamic~glaze} with minimal ingredients---ideally just grape must and wine vinegar
    \item Avoid products with added thickeners or caramel coloring
    \item European-style \textbf{butter} provides richer flavor due to higher butterfat content
    \item Smaller \textbf{sprouts} (1-1½ inch) can be halved; larger ones should be quartered for even cooking
\end{itemize}

\subsection*{Preparation Tips}
\begin{itemize}
    \item Loose outer leaves will char more aggressively---this is desirable for texture and bitter complexity
    \item Don't be afraid of significantly blackened edges---this char is where the flavor complexity develops
    \item The charred leaves provide bitter, smoky notes that balance the sweet-tart \textbf{balsamic} and rich \textbf{butter}
    \item The transition from perfect brown \textbf{butter} to burnt happens in seconds---remove from heat when you smell nuttiness and see amber color
    \item Residual heat will continue cooking the \textbf{butter} briefly after removal from heat
    \item The \textbf{balsamic} and \textbf{butter} mixture will remain somewhat separated---don't over-whisk trying to force an emulsion
    \item Toss \textbf{sprouts} gently to avoid knocking off the crispy charred leaves
\end{itemize}

\subsection*{Make Ahead \& Storage}
\begin{itemize}
    \item \textbf{Brussels~sprouts} can be trimmed and halved up to \textit{24~hours} in advance
    \item Store prepared \textbf{sprouts} in an airtight container in the refrigerator
    \item The brown \textbf{butter} sauce must be made fresh and served immediately
    \item This dish does not reheat well---the crispy texture is lost
    \item Best enjoyed immediately after preparation
\end{itemize}

\subsection*{Serving Suggestions}
\begin{itemize}
    \item Serve as a side dish for roasted meats, particularly pork or beef
    \item Pairs well with rich, fatty main courses that benefit from acidic contrast
    \item The char's bitterness, \textbf{balsamic's} acidity, and brown \textbf{butter's} richness create a three-point flavor system
    \item The char provides textural contrast---crispy, almost chip-like leaves against tender interiors
    \item Consider garnishing with additional fresh \textbf{rosemary} leaves if desired
    \item For larger batches, cook \textbf{sprouts} in multiple air fryer batches and keep finished \textbf{sprouts} warm in a \textit{200°F} oven while preparing subsequent batches
\end{itemize}
