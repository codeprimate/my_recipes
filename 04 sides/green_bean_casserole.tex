\section*{Ingredients}
\setlength{\columnsep}{20pt}
\begin{multicols}{2}
\noindent
    Green beans, fresh \dotfill 1\nicefrac{1}{2} lbs. \\
    Butter \dotfill \nicefrac{1}{2} cup \\
    Mushrooms, fresh \dotfill \nicefrac{1}{2} lb. \\
    Onion, small \dotfill 1 \\
    All-purpose flour \dotfill \nicefrac{1}{4} cup \\
    Whole milk \dotfill 1\nicefrac{1}{2} cups \\
    \columnbreak
    Sour cream \dotfill 1 cup \\
    Cheddar cheese, shredded \dotfill 1 cup \\
    Salt \dotfill 1 tsp. \\
    Black pepper \dotfill 1 tsp. \\
    French-fried onions \dotfill 2 cups
\end{multicols}

\section*{Directions}

\noindent
Preheat oven to \textit{350Â°F} ---
Trim \textbf{green beans}; set aside in \textit{Large Bowl~\#1} ---
Slice \textbf{mushrooms}; set aside in \textit{Medium Bowl~\#1} ---
Chop \textbf{onion}; set aside in \textit{Small Bowl~\#1} ---
Shred \textbf{cheddar cheese}; set aside in \textit{Medium Bowl~\#2} ---
Grease a 2\nicefrac{1}{2}~quart baking dish

\begin{enumerate}
    \item Bring a large pot of water to a boil. Add \textbf{green beans} (\textit{Large Bowl~\#1}) and blanch for \textit{5~minutes}. Drain and immediately plunge into ice water. Drain again and return to \textit{Large Bowl~\#1}.
    
    \item In a large skillet, melt \nicefrac{1}{2}~cup \textbf{butter} over medium heat. Add \textbf{mushrooms} (\textit{Medium Bowl~\#1}) and \textbf{onion} (\textit{Small Bowl~\#1}); cook for \textit{5~minutes}, until tender.
    
    \item Stir in \nicefrac{1}{4}~cup \textbf{flour} and cook for \textit{2~minutes}, stirring constantly.
    
    \item Gradually stir in 1\nicefrac{1}{2}~cups \textbf{milk} and cook for \textit{2-3~minutes}, until mixture thickens.
    
    \item Stir in 1~cup \textbf{sour cream}, \textbf{cheese} (1~cup, \textit{Medium Bowl~\#2}), 1~tsp. \textbf{salt}, 1~tsp. \textbf{pepper}, and blanched \textbf{green beans} (\textit{Large Bowl~\#1}).
    
    \item Transfer mixture to prepared baking dish. Bake for \textit{25~minutes}.
    
    \item Sprinkle evenly with 2~cups \textbf{French-fried onions}. Bake for \textit{15~minutes} longer, until hot and bubbly.
\end{enumerate}
