\section*{Ingredients}
\setlength{\columnsep}{20pt}
\begin{multicols}{2}
\noindent
    English cucumber \dotfill 1 large \\
    \textit{or Regular cucumbers \dotfill 2 medium }\\
    Kosher salt \dotfill 1\nicefrac{1}{2} tsp. \\
    Garlic cloves \dotfill 4 \\
    Green onions \dotfill 2 \\
    Gochugaru \dotfill 1\nicefrac{1}{2} tsp. \\
    \columnbreak
    Rice vinegar \dotfill 1 Tbsp. \\
    Toasted sesame oil \dotfill 2 tsp. \\
    Sugar \dotfill \nicefrac{1}{2} tsp. \\
    Toasted sesame seeds \dotfill 1 Tbsp. \\
    Black pepper \dotfill \nicefrac{1}{4} tsp. \\
\end{multicols}

\section*{Directions}

\noindent
If using regular cucumbers, peel and halve lengthwise ---
Mince \textbf{garlic}; set aside in \textit{Small Bowl~\#1} ---
Slice \textbf{green onions} thinly; set aside in \textit{Small Bowl~\#2} ---
Prepare ice water bath

\begin{enumerate}
    \item For regular cucumbers: Remove seeds with a spoon, then slice into \nicefrac{1}{8}~inch half-moons. For English cucumber: Slice into \nicefrac{1}{8}~inch rounds. Set aside in \textit{Large Bowl~\#1}.
    
    \item Place cucumber slices (\textit{Large Bowl~\#1}) in a colander, sprinkle with 1\nicefrac{1}{2}~tsp. \textbf{kosher salt}, and toss gently. Let drain for \textit{30~minutes} (regular) or \textit{20~minutes} (English).
    
    \item Rinse cucumbers briefly under cold water. Plunge into ice water bath for \textit{1~minute}. Drain well and pat thoroughly dry with paper towels. Return to \textit{Large Bowl~\#1}.
    
    \item In \textit{Medium Bowl~\#1}, combine minced \textbf{garlic} (\textit{Small Bowl~\#1}), 1\nicefrac{1}{2}~tsp. \textbf{gochugaru}, 1~Tbsp. \textbf{rice vinegar}, 2~tsp. \textbf{toasted sesame oil}, \nicefrac{1}{2}~tsp. \textbf{sugar}, and \nicefrac{1}{4}~tsp. \textbf{black pepper}.
    
    \item Add dried cucumber slices (\textit{Large Bowl~\#1}) and sliced \textbf{green onions} (\textit{Small Bowl~\#2}) to the bowl. Toss gently but thoroughly to coat.
    
    \item Sprinkle with 1~Tbsp. \textbf{toasted sesame seeds}, toss once more, and let rest for \textit{10~minutes} before serving.
\end{enumerate}

\subsection*{Equipment Required}
\begin{itemize}
    \item Large colander
    \item Large mixing bowl
    \item Sharp knife and cutting board
    \item Paper towels
    \item Measuring spoons
    \item Large bowl for ice bath
    \item Vegetable peeler (if using regular cucumbers)
\end{itemize}

\subsection*{Mise en Place}
\begin{itemize}
    \item Prepare ice water bath before starting
    \item Measure all seasonings before beginning
    \item Have paper towels ready for drying cucumbers
    \item Toast \textbf{sesame seeds} if needed
\end{itemize}

\subsection*{Ingredient Tips}
\begin{itemize}
    \item English cucumbers are ideal, but regular cucumbers work well when properly prepared
    \item Fresh \textbf{garlic} is essential - do not substitute pre-minced
    \item \textbf{Gochugaru} can be found at Asian markets - no good substitute exists
    \item Use unseasoned \textbf{rice vinegar} for better control of flavors
    \item Regular table salt can be used, but reduce quantity by half
\end{itemize}

\subsection*{Preparation Tips}
\begin{itemize}
    \item Slice cucumbers as uniformly as possible for even texture
    \item Don't skip the ice bath - it ensures maximum crispness
    \item Pat cucumbers very dry before adding seasonings
    \item Adjust \textbf{gochugaru} amount based on desired heat level
    \item Test seasoning before final rest period
\end{itemize}

\subsection*{Make Ahead \& Storage}
\begin{itemize}
    \item Best served fresh, within \textit{2~hours} of preparation
    \item Can be made up to \textit{4~hours} ahead, but texture will soften
    \item Store leftovers in airtight container for up to \textit{24~hours}
    \item Do not freeze
    \item Drain any accumulated liquid before serving leftover portions
\end{itemize}

\subsection*{Serving Suggestions}
\begin{itemize}
    \item Perfect accompaniment to Korean BBQ or grilled meats
    \item Serve cold or at room temperature
    \item Excellent as part of a banchan (Korean side dish) spread
    \item Can be used as a sandwich or burger topping
    \item Pairs well with rice dishes and noodles
\end{itemize}
