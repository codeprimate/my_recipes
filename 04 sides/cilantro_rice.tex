\section*{Ingredients}
\setlength{\columnsep}{20pt}
\begin{multicols}{2}
\noindent
    Long-grain white rice \dotfill 2 cups \\
    Chicken broth \dotfill 3 cups \\
    Olive oil \dotfill 2 Tbsp. \\
    Onion, small \dotfill 1 \\
    Garlic cloves \dotfill 3 \\
    Salt \dotfill 1 tsp. \\
    \columnbreak
    Lime juice, fresh \dotfill \nicefrac{1}{4} cup \\
    Lime zest \dotfill 1 lime \\
    Cilantro, fresh \dotfill \nicefrac{1}{2} cup \\
    Butter \dotfill 2 Tbsp. \\
    Jalapeño (optional) \dotfill 1 Tbsp. \\
\end{multicols}

\section*{Directions}

\noindent
Rinse \textbf{rice} until water runs clear ---
Finely dice \textbf{onion}; set aside in \textit{Small Bowl~\#1} ---
Mince \textbf{garlic}; set aside in \textit{Small Bowl~\#2} ---
Zest and juice \textbf{limes}; set zest in \textit{Small Bowl~\#3} and juice in \textit{Small Bowl~\#4} ---
Finely chop \textbf{cilantro}; set aside in \textit{Small Bowl~\#5} ---
If using, finely dice \textbf{jalapeño}; set aside in \textit{Small Bowl~\#6}

\begin{enumerate}
    \item Heat 2~Tbsp. \textbf{olive oil} in a medium saucepan over medium heat. Sauté \textbf{onion} (\textit{Small Bowl~\#1}) for \textit{2-3~minutes} until translucent.
    \item Add \textbf{garlic} (\textit{Small Bowl~\#2}) and optional \textbf{jalapeño} (\textit{Small Bowl~\#6}) and cook for \textit{30~seconds} until fragrant.
    \item Add rinsed \textbf{rice} to the pan and stir to coat with oil. Toast for about \textit{2~minutes}, stirring frequently.
    \item Transfer rice mixture to rice cooker. Add 3~cups \textbf{chicken broth} to appropriate fill line. Stir gently to combine.
    \item Close lid and cook on standard white rice setting until cycle is complete.
    \item Once done, let rice sit for an additional \textit{5-10~minutes} with the lid closed.
    \item Open lid and fluff rice with a fork. Add \textbf{lime juice} (\nicefrac{1}{4}~cup, \textit{Small Bowl~\#4}), \textbf{lime zest} (\textit{Small Bowl~\#3}), chopped \textbf{cilantro} (\nicefrac{1}{2}~cup, \textit{Small Bowl~\#5}), and 2~Tbsp. \textbf{butter}. Fold gently until well combined and butter is melted.
    \item Taste and adjust seasoning, if needed, with salt, lime juice, or cilantro to preference.
\end{enumerate}
